Sample: ProductCard Component

components/product/ProductCard.tsx 
import { useState } from "react";
import Image from "next/image";
import { useCartStore } from "@/stores/cart";
import { useWishlist } from "@/hooks/useWishlist";
import { formatCurrency } from "@/lib/utils";
import type { Product } from "@ecom/types";

interface ProductCardProps {
  product: Product;
  priority?: boolean;
}

export function ProductCard({ product, priority = false }: ProductCardProps) {
  const [isAdding, setIsAdding] = useState(false);
  const addItem = useCartStore(s => s.addItem);
  const { isWishlisted, toggle } = useWishlist(product.id);

  async function handleAddToCart() {
    setIsAdding(true);
    await addItem(product);
    setIsAdding(false);
  }

  return (
    <article className="group relative bg-white rounded-2xl overflow-hidden">
      <div className="relative aspect-square">
        <Image
          src={product.images[0].url}
          alt={product.name}
          fill className="object-cover group-hover:scale-105 transition-transform"
          placeholder="blur" blurDataURL={product.images[0].blurData}
          priority={priority}
        />
        <button onClick={() => toggle()} className="absolute top-3 right-3 p-2">
          {isWishlisted ? "♥" : "♡"}
        </button>
        {product.discountPercent > 0 && (
          <span className="absolute top-3 left-3 badge-sale">
            -{product.discountPercent}%
          </span>
        )}
      </div>

      <div className="p-4">
        <p className="text-sm text-gray-500">{product.brand}</p>
        <h3 className="font-semibold mt-1">{product.name}</h3>
        <div className="flex items-center justify-between mt-2">
          <div>
            <span className="font-bold text-lg">
              {formatCurrency(product.salePrice)}
            </span>
            {product.salePrice < product.price && (
              <span className="line-through text-gray-400 ml-2 text-sm">
                {formatCurrency(product.price)}
              </span>
            )}
          </div>
          <button
            onClick={handleAddToCart} disabled={isAdding}
            className="btn-primary text-sm"
          >
            {isAdding ? "Adding..." : "Add to Cart"}
          </button>
        </div>
      </div>
    </article>
  );
}

06 State Management

Zustand for global UI state (cart, auth, UI); React Query for server state (products, orders).

Cart Store (Zustand + Persist)

stores/cart.ts 
import { create } from "zustand";
import { persist } from "zustand/middleware";
import type { CartItem, Product } from "@ecom/types";

interface CartState {
  items: CartItem[];
  coupon: string | null;
  addItem: (product: Product, qty?: number) => void;
  removeItem: (productId: string) => void;
  updateQty: (productId: string, qty: number) => void;
  applyCoupon: (code: string) => void;
  clear: () => void;
  total: () => number;
}

export const useCartStore = create<CartState>()(
  persist(
    (set, get) => ({
      items: [],
      coupon: null,

      addItem: (product, qty = 1) => set(state => {
        const existing = state.items.find(i => i.productId === product.id);
        if (existing) {
          return {
            items: state.items.map(i =>
              i.productId === product.id ? { ...i, qty: i.qty + qty } : i
            )
          };
        }
        return {
          items: [...state.items, {
            productId: product.id,
            name: product.name,
            price: product.salePrice,
            image: product.images[0].url,
            qty,
          }]
        };
      }),

      removeItem: (productId) => set(s => ({
        items: s.items.filter(i => i.productId !== productId)
      })),

      updateQty: (productId, qty) => set(s => ({
        items: qty <= 0
          ? s.items.filter(i => i.productId !== productId)
          : s.items.map(i => i.productId === productId ? {...i, qty} : i)
      })),

      applyCoupon: (code) => set({ coupon: code }),
      clear: () => set({ items: [], coupon: null }),

      total: () => get().items.reduce(
        (sum, i) => sum + i.price * i.qty, 0
      ),
    }),
    { name: "ecom-cart", version: 1 }
  )
);

08 Authentication & Authorization

JWT-based auth with short-lived access tokens (15m) and long-lived refresh tokens (7d) stored in httpOnly cookies.

🔑
JWT Access Token
15-minute expiry. Stored in memory. Signed with RS256. Contains userId, email, role.
🔄
Refresh Token
7-day expiry. httpOnly + Secure cookie. Rotated on each use. Stored hashed in DB.
🌐
OAuth 2.0
Google, Facebook login via Passport.js. Auto-links accounts with same email.
🛡
RBAC
Roles: customer, vendor, admin, super-admin. Permissions checked at middleware level.
Security note Never store JWTs in localStorage. Access tokens live in React state; refresh tokens in httpOnly cookies only accessible server-side.

Auth Middleware

shared/middleware/authenticate.ts 
import jwt from "jsonwebtoken";
import { UnauthorizedException } from "../exceptions";

export function authenticate(req, res, next) {
  const authHeader = req.headers["authorization"];
  if (!authHeader?.startsWith("Bearer "))
    throw new UnauthorizedException("Missing token");

  const token = authHeader.slice(7);
  try {
    req.user = jwt.verify(token, process.env.JWT_PUBLIC_KEY, {
      algorithms: ["RS256"],
    });
    next();
  } catch {
    throw new UnauthorizedException("Invalid or expired token");
  }
}

export function authorize(...roles: string[]) {
  return (req, res, next) => {
    if (!roles.includes(req.user.role))
      throw new ForbiddenException("Insufficient permissions");
    next();
  };
}

Design Decisions & Patterns

📸
Snapshot Pattern on OrderItem
Product name, SKU, image, and price are copied into OrderItem at creation time. This means historical orders remain accurate even if the product is later edited or deleted.
🗑
Soft Deletes on User & Product
deletedAt timestamp instead of hard delete. Protects referential integrity for orders. All queries filter WHERE deletedAt IS NULL. Prisma middleware adds this automatically.
🌲
Self-Referential Category Tree
parentId FK back to the same table allows unlimited nesting depth. The level field is denormalised for fast top-level queries without recursive CTEs.
🪙
Decimal for All Money Fields
DECIMAL(10,2) instead of Float — floats can't represent ₹0.10 accurately. The app converts to paise (integer) only when calling Stripe.
👤
Guest Cart via sessionId
The Cart table supports both userId (logged-in) and sessionId (guest). On login, guest cart is merged into the user's cart, then the session cart is deleted.
🔒
Payment Data Isolation
Stripe card details never touch your DB — only the paymentIntentId and raw Stripe event JSON are stored. The Payment table is kept separate from Order for clean separation.

10 Payment Integration (Stripe)

1. Create PaymentIntent on server
When user clicks "Pay", call POST /api/v1/payments/intent which creates a Stripe PaymentIntent and returns the client_secret.
2. Confirm on client with Stripe Elements
Use stripe.confirmPayment() with the client_secret. Handles card, Apple Pay, Google Pay, UPI automatically.
3. Webhook confirms payment
Stripe sends payment_intent.succeeded to your webhook. Verify signature, then mark order as PAID and trigger fulfilment.
4. Fulfilment pipeline
Webhook triggers Bull job: reduce inventory → send confirmation email → notify vendor → update order status.
modules/payments/payments.service.ts 
import Stripe from "stripe";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);

export async function createPaymentIntent(orderId: string) {
  const order = await db.order.findUniqueOrThrow({ where: { id: orderId } });

  const intent = await stripe.paymentIntents.create({
    amount: toStripeAmount(order.total),  // paise/cents
    currency: "inr",
    metadata: { orderId, userId: order.userId },
    payment_method_types: ["card", "upi"],
    receipt_email: order.user.email,
  });

  await db.order.update({
    where: { id: orderId },
    data: { paymentId: intent.id, status: "PAYMENT_PENDING" }
  });

  return { clientSecret: intent.client_secret };
}

export async function handleWebhook(rawBody: Buffer, sig: string) {
  const event = stripe.webhooks.constructEvent(
    rawBody, sig, process.env.STRIPE_WEBHOOK_SECRET!
  );

  switch (event.type) {
    case "payment_intent.succeeded":
      await fulfillOrder(event.data.object.metadata.orderId);
      break;
    case "payment_intent.payment_failed":
      await handlePaymentFailed(event.data.object.metadata.orderId);
      break;
  }
}

export async function searchProducts(params: SearchParams) {
  const { q, category, minPrice, maxPrice, brand, rating, sort, page } = params;

  const query = {
    index: "products",
    body: {
      query: {
        bool: {
          must: q ? [{ multi_match: { query: q,
            fields: ["name^3", "description", "tags^2", "brand.name"],
            fuzziness: "AUTO"
          }}] : [{ match_all: {} }],
          filter: [
            category && { term: { "category.slug": category } },
            minPrice && { range: { salePrice: { gte: minPrice, lte: maxPrice } } },
            brand && { term: { "brand.slug": brand } },
            rating && { range: { avgRating: { gte: rating } } },
          ].filter(Boolean),
        }
      },
      aggs: {
        categories: { terms: { field: "category.slug" } },
        brands:     { terms: { field: "brand.slug" } },
        price_range:{ stats: { field: "salePrice" } },
        avg_rating: { avg: { field: "avgRating" } },
      },
      sort: buildSort(sort),
      from: (page - 1) * 20, size: 20,
    }
  };

  return esClient.search(query);
}

13 File Storage & CDN

AWS S3 for origin storage. CloudFront as CDN for images and static assets. Images processed on upload with Sharp.

shared/utils/storage.ts 
import sharp from "sharp";
import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";

const s3 = new S3Client({ region: process.env.AWS_REGION });

const SIZES = [
  { suffix: "thumb", width: 200, height: 200 },
  { suffix: "medium", width: 600, height: 600 },
  { suffix: "large", width: 1200, height: 1200 },
];

export async function uploadProductImage(
  file: Buffer,
  productId: string
): Promise<ImageUrls> {
  const urls: Record<string, string> = {};

  for (const { suffix, width, height } of SIZES) {
    const webp = await sharp(file)
      .resize(width, height, { fit: "cover" })
      .webp({ quality: 85 })
      .toBuffer();

    const key = `products/${productId}/${suffix}.webp`;

    await s3.send(new PutObjectCommand({
      Bucket: process.env.S3_BUCKET!,
      Key: key,
      Body: webp,
      ContentType: "image/webp",
      CacheControl: "public, max-age=31536000, immutable",
    }));

    urls[suffix] = `${process.env.CDN_URL}/${key}`;
  }

  return { ...urls, blurData: await generateBlurPlaceholder(file) };
}

22 Security Checklist

Input Validation — Zod on every endpoint
All request bodies, query params, and path params validated with Zod schemas. Unknown fields stripped. SQL injected strings rejected.
OWASP Headers — Helmet.js
Content-Security-Policy, X-Frame-Options, X-Content-Type-Options, HSTS, Referrer-Policy all set via Helmet middleware.
Rate Limiting — per IP + per user
Auth endpoints: 5 attempts/min. API: 100 req/min per IP. Stripe webhooks: IP whitelist only.
CSRF Protection — SameSite cookies
Refresh tokens use SameSite=Strict cookies. Double-submit CSRF tokens on state-changing forms.
Secrets Management — AWS Secrets Manager
All secrets in AWS Secrets Manager in production. Never hardcode. Rotate Stripe keys quarterly.
Dependency Scanning — Dependabot + Snyk
Auto PRs for vulnerable packages. Block merging if HIGH severity vulnerability detected in CI.

26 Inventory Management

Race Condition Warning Always use database transactions with SELECT FOR UPDATE when decrementing stock to prevent overselling. Never check stock and update in separate queries.
modules/inventory/inventory.service.ts 
export async function reserveStock(
  items: OrderItem[],
  tx: PrismaTransaction
) {
  for (const item of items) {
    // Pessimistic lock — prevents race conditions
    const product = await tx.$queryRaw`
      SELECT id, stock FROM products
      WHERE id = ${item.productId}
      FOR UPDATE
    `;

    if (product.stock < item.qty) {
      throw new BadRequestException(
        `Insufficient stock for ${product.name}`
      );
    }

    await tx.product.update({
      where: { id: item.productId },
      data: {
        stock: { decrement: item.qty },
        soldCount: { increment: item.qty },
      }
    });
  }
}
