import React, { useState, useEffect } from 'react';
import { fetchTasks, createTask, deleteTask } from './api';
import { getCurrentUser, signOut } from './auth';

export default function App() {
  const [tasks, setTasks] = useState([]);
  const [user, setUser]   = useState(null);
  const [input, setInput] = useState('');
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Get logged-in user from Cognito JWT
    getCurrentUser().then(u => {
      setUser(u);
      if (u) loadTasks();
    });
  }, []);

  async function loadTasks() {
    setLoading(true);
    const data = await fetchTasks(); // GET /tasks via API Gateway
    setTasks(data);
    setLoading(false);
  }

  async function addTask() {
    if (!input.trim()) return;
    const task = await createTask({ title: input }); // POST /tasks
    setTasks(prev => [task, ...prev]);
    setInput('');
  }

  async function removeTask(id) {
    await deleteTask(id); // DELETE /tasks/{id}
    setTasks(prev => prev.filter(t => t.id !== id));
  }

  if (!user) return <LoginPage />;

  return (
    <div className="app">
      <header>
        <h1>Task Manager</h1>
        <button onClick={signOut}>Logout</button>
      </header>

      <div className="input-row">
        <input
          value={input}
          onChange={e => setInput(e.target.value)}
          placeholder="Add a task..."
          onKeyDown={e => e.key === 'Enter' && addTask()}
        />
        <button onClick={addTask}>Add Task</button>
      </div>

      {loading ? <p>Loading...</p> : (
        <ul>
          {tasks.map(task => (
            <li key={task.id}>
              {task.title}
              <button onClick={() => removeTask(task.id)}>✕</button>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
src/api.js — API Gateway Client 
import { getJwtToken } from './auth';

const API_BASE = process.env.REACT_APP_API_URL;
// e.g. https://abc123.execute-api.ap-south-1.amazonaws.com/prod

async function authFetch(path, options = {}) {
  const token = await getJwtToken(); // From Cognito
  const res = await fetch(`${API_BASE}${path}`, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`, // JWT sent to API Gateway
      ...options.headers,
    },
  });
  if (!res.ok) throw new Error(await res.text());
  return res.json();
}

export const fetchTasks  = ()     => authFetch('/tasks');
export const createTask  = (body)  => authFetch('/tasks', { method: 'POST',   body: JSON.stringify(body) });
export const updateTask  = (id, b) => authFetch(`/tasks/${id}`, { method: 'PUT',    body: JSON.stringify(b) });
export const deleteTask  = (id)    => authFetch(`/tasks/${id}`, { method: 'DELETE' });
export const uploadFile  = (data)  => authFetch('/upload',   { method: 'POST',   body: data });

const express       = require('express');
const cors          = require('cors');
const { Pool }      = require('pg');         // RDS PostgreSQL
const redis         = require('redis');       // ElastiCache
const AWS           = require('aws-sdk');
const { verifyJWT } = require('./auth');

const app  = express();
const sqs  = new AWS.SQS();
const s3   = new AWS.S3();
const ses  = new AWS.SES({ region: 'ap-south-1' });
const ssm  = new AWS.SecretsManager();

app.use(cors());
app.use(express.json());

// ─── Database (RDS PostgreSQL) ─────────────────────────────
const pool = new Pool({
  host:     process.env.DB_HOST,     // RDS endpoint (from Secrets Manager)
  database: process.env.DB_NAME,
  user:     process.env.DB_USER,
  password: process.env.DB_PASS,
  port:     5432,
  ssl:      { rejectUnauthorized: false },
});

// ─── Redis Cache (ElastiCache) ─────────────────────────────
const redisClient = redis.createClient({ url: process.env.REDIS_URL });
await redisClient.connect();

// ─── Auth Middleware ──────────────────────────────────────
async function authenticate(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'Unauthorized' });
  try {
    req.user = await verifyJWT(token); // Validates Cognito JWT
    next();
  } catch { res.status(401).json({ error: 'Invalid token' }); }
}

// ─── GET /tasks ───────────────────────────────────────────
app.get('/tasks', authenticate, async (req, res) => {
  const cacheKey = `tasks:${req.user.sub}`;
  const cached   = await redisClient.get(cacheKey); // Check Redis first
  if (cached) return res.json(JSON.parse(cached));

  const { rows } = await pool.query(
    'SELECT * FROM tasks WHERE user_id=$1 ORDER BY created_at DESC',
    [req.user.sub]
  );
  await redisClient.setEx(cacheKey, 60, JSON.stringify(rows)); // Cache 60s
  res.json(rows);
});

// ─── POST /tasks ──────────────────────────────────────────
app.post('/tasks', authenticate, async (req, res) => {
  const { title, description, due_date } = req.body;
  const { rows } = await pool.query(
    'INSERT INTO tasks(user_id,title,description,due_date) VALUES($1,$2,$3,$4) RETURNING *',
    [req.user.sub, title, description, due_date]
  );
  await redisClient.del(`tasks:${req.user.sub}`); // Invalidate cache

  // Push to SQS for async processing (email notification)
  await sqs.sendMessage({
    QueueUrl: process.env.SQS_TASK_QUEUE_URL,
    MessageBody: JSON.stringify({ type: 'TASK_CREATED', task: rows[0], userId: req.user.sub })
  }).promise();

  res.status(201).json(rows[0]);
});

// ─── S3 Presigned URL for file upload ─────────────────────
app.post('/upload', authenticate, async (req, res) => {
  const { filename, contentType } = req.body;
  const key = `uploads/${req.user.sub}/${Date.now()}-${filename}`;
  const url = s3.getSignedUrl('putObject', {
    Bucket:      process.env.S3_UPLOADS_BUCKET,
    Key:         key,
    ContentType: contentType,
    Expires:     300, // 5 minutes
  });
  res.json({ uploadUrl: url, key });
});

app.listen(3000, () => console.log('Server running on port 3000'));
lambda/emailWorker.js — SQS Consumer Lambda 
const AWS = require('aws-sdk');
const ses = new AWS.SES({ region: 'ap-south-1' });

// Triggered automatically when SQS has messages
exports.handler = async (event) => {
  for (const record of event.Records) {
    const { type, task, userId } = JSON.parse(record.body);

    if (type === 'TASK_CREATED') {
      // Look up user email from DynamoDB
      const dynamo = new AWS.DynamoDB.DocumentClient();
      const { Item: user } = await dynamo.get({
        TableName: process.env.USERS_TABLE,
        Key: { userId }
      }).promise();

      // Send confirmation email via SES
      await ses.sendEmail({
        Source:      'noreply@yourdomain.com',
        Destination: { ToAddresses: [user.email] },
        Message: {
          Subject: { Data: `✅ Task Created: ${task.title}` },
          Body:    { Text: { Data: `Your task "${task.title}" was created successfully!` } }
        }
      }).promise();
    }
  }
  return { statusCode: 200 };
};

db/schema.sql — RDS PostgreSQL Schema 
-- Run this in your RDS PostgreSQL instance

CREATE TABLE users (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cognito_id VARCHAR(128) UNIQUE NOT NULL,   -- Cognito sub
  email      VARCHAR(255) UNIQUE NOT NULL,
  name       VARCHAR(100),
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE projects (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     UUID REFERENCES users(id) ON DELETE CASCADE,
  name        VARCHAR(200) NOT NULL,
  description TEXT,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE tasks (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     VARCHAR(128) NOT NULL,          -- Cognito sub
  project_id  UUID REFERENCES projects(id),
  title       VARCHAR(500) NOT NULL,
  description TEXT,
  status      VARCHAR(20) DEFAULT 'pending',   -- pending/done/archived
  priority    SMALLINT DEFAULT 2,              -- 1=low,2=med,3=high
  due_date    TIMESTAMPTZ,
  s3_file_key VARCHAR(500),                    -- Attached file in S3
  created_at  TIMESTAMPTZ DEFAULT now(),
  updated_at  TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_tasks_user    ON tasks(user_id);
CREATE INDEX idx_tasks_status  ON tasks(status);
CREATE INDEX idx_tasks_due     ON tasks(due_date);

-- Auto-update updated_at on row change
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_updated_at
  BEFORE UPDATE ON tasks
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();

#!/bin/bash
# ── 1. Create VPC ───────────────────────────────────────────
VPC_ID=$(aws ec2 create-vpc --cidr-block 10.0.0.0/16 \
  --query 'Vpc.VpcId' --output text)
aws ec2 create-tags --resources $VPC_ID \
  --tags Key=Name,Value=taskapp-vpc

# ── 2. Create Subnets ───────────────────────────────────────
# Public subnet (for ALB)
PUB_SUBNET=$(aws ec2 create-subnet --vpc-id $VPC_ID \
  --cidr-block 10.0.1.0/24 --availability-zone ap-south-1a \
  --query 'Subnet.SubnetId' --output text)

# Private subnet (for EC2, RDS, ElastiCache)
PRV_SUBNET=$(aws ec2 create-subnet --vpc-id $VPC_ID \
  --cidr-block 10.0.2.0/24 --availability-zone ap-south-1b \
  --query 'Subnet.SubnetId' --output text)

# ── 3. Internet Gateway ─────────────────────────────────────
IGW_ID=$(aws ec2 create-internet-gateway \
  --query 'InternetGateway.InternetGatewayId' --output text)
aws ec2 attach-internet-gateway --vpc-id $VPC_ID \
  --internet-gateway-id $IGW_ID

# ── 4. Launch EC2 (t3.micro — Free Tier) ───────────────────
aws ec2 run-instances \
  --image-id ami-0c55b159cbfafe1f0 \
  --instance-type t3.micro \
  --subnet-id $PRV_SUBNET \
  --iam-instance-profile Name=taskapp-ec2-role \
  --user-data file://userdata.sh \
  --tag-specifications 'ResourceType=instance,Tags=[{Key=Name,Value=taskapp-server}]'

WAF — Blocks SQLi, XSS, bad bots at CloudFront edge before reaching your server

Section 10
Step-by-Step Deployment Guide
Follow these steps in order to deploy the complete stack
1

AWS Account & CLI Setup

Create an AWS account, create an IAM user with AdministratorAccess, and configure the CLI.

aws configure
# Enter: Access Key, Secret Key, Region (ap-south-1), Output (json)
2

Create VPC + Subnets + Security Groups

Run the VPC setup script. Creates public subnets (ALB), private subnets (EC2, RDS), and a NAT Gateway for private subnet internet access.

bash infra/setup.sh
3

Setup Cognito User Pool

Create a Cognito User Pool with email sign-in, MFA optional, and an App Client. Note the User Pool ID and App Client ID.

aws cognito-idp create-user-pool --pool-name taskapp-users --auto-verified-attributes email
4

Launch RDS PostgreSQL

Create a db.t3.micro RDS instance in the private subnet. Multi-AZ for production, single-AZ for dev. Automated backups enabled.

aws rds create-db-instance --db-instance-identifier taskapp-db --db-instance-class db.t3.micro --engine postgres --master-username admin --master-user-password <use-secrets-manager> --allocated-storage 20 --vpc-security-group-ids sg-xxx --db-subnet-group-name taskapp-subnet-group
5

Create ElastiCache Redis Cluster

Deploy a cache.t3.micro Redis node in the private subnet. Must be in the same VPC as EC2.

aws elasticache create-cache-cluster --cache-cluster-id taskapp-redis --engine redis --cache-node-type cache.t3.micro --num-cache-nodes 1 --cache-subnet-group-name taskapp-cache-subnet
6

Store Secrets in Secrets Manager

Never hardcode credentials. Store DB password, Redis URL, JWT secret here. EC2/Lambda fetch at runtime via IAM role.

aws secretsmanager create-secret --name taskapp/db --secret-string '{"host":"xxx.rds.amazonaws.com","user":"admin","password":"Str0ngP@ss"}'
7

Create SQS Queue + SNS Topic

The task queue receives messages from the API and triggers Lambda to send emails. SNS for broadcast notifications.

aws sqs create-queue --queue-name taskapp-tasks --attributes VisibilityTimeout=60,MessageRetentionPeriod=86400
aws sns create-topic --name taskapp-notifications
8

Launch EC2 + Application Load Balancer

Launch EC2 (t3.micro) with the userdata bootstrap script. Create ALB in public subnets forwarding to EC2 on port 3000.

bash infra/launch-ec2.sh
bash infra/create-alb.sh
9

Deploy Lambda Functions

Package and deploy the email worker Lambda. Set the SQS queue as trigger with batch size 10.

cd lambda && zip -r function.zip .
aws lambda create-function --function-name taskapp-email-worker --runtime nodejs18.x --zip-file fileb://function.zip --role arn:aws:iam::ACCOUNT:role/taskapp-lambda-role --handler emailWorker.handler
10

Setup API Gateway

Create REST API, add Cognito authorizer, create resources (/tasks, /upload), and link methods to the EC2 ALB endpoint.

aws apigateway create-rest-api --name taskapp-api --region ap-south-1
11

S3 Bucket + CloudFront for Frontend

Create S3 bucket for React app. Configure static website hosting. Create CloudFront distribution pointing to S3 with ACM HTTPS cert.

aws s3 mb s3://taskapp-frontend-YOUR-UNIQUE-NAME --region ap-south-1
aws s3 website s3://taskapp-frontend-YOUR-UNIQUE-NAME --index-document index.html
12

Route 53 + ACM Certificate

Request an ACM certificate for your domain. Validate via DNS. Create Route 53 A record aliasing to CloudFront.

aws acm request-certificate --domain-name yourdomain.com --validation-method DNS --region us-east-1
13

Setup CI/CD Pipeline

Connect GitHub repo to CodePipeline. Add buildspec.yml. Pipeline auto-triggers on every push to main branch.

aws codepipeline create-pipeline --cli-input-json file://pipeline.json
14

CloudWatch Alarms + WAF

Set up CloudWatch alarms for CPU, errors, RDS storage. Attach WAF to CloudFront. Enable CloudTrail for audit logging.

bash monitoring/alarms.sh
bash security/waf-setup.sh
15

✅ Deploy & Verify

Build and push React app to S3. Verify CloudFront serves the app. Test API Gateway endpoints. Check CloudWatch for logs.

npm run build && aws s3 sync ./build s3://taskapp-frontend-YOUR-UNIQUE-NAME --delete
curl https://api.yourdomain.com/health
