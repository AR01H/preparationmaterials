Lambda Code — lambda_function.py
lambda_function.pyPython
import boto3, json, urllib.request from datetime import datetime # Clients secrets = boto3.client('secretsmanager') db = boto3.resource('dynamodb').Table('weather_history') def get_api_key(): r = secrets.get_secret_value(SecretId='weather/apikey') return json.loads(r['SecretString'])['weather_api_key'] def lambda_handler(event, ctx): city = event.get('queryStringParameters', {}).get('city', 'Mumbai') # Fetch weather api_key = get_api_key() url = (f'https://api.openweathermap.org/data/2.5/weather' f'?q={city}&appid={api_key}&units=metric') with urllib.request.urlopen(url) as resp: data = json.loads(resp.read()) result = { 'city': data['name'], 'temp_c': data['main']['temp'], 'feels_like': data['main']['feels_like'], 'humidity': data['main']['humidity'], 'description': data['weather'][0]['description'], 'wind_speed': data['wind']['speed'] } # Save to DynamoDB history db.put_item(Item={ 'city': result['city'], 'timestamp': datetime.utcnow().isoformat(), 'data': json.dumps(result) }) return { 'statusCode': 200, 'headers': {'Access-Control-Allow-Origin': '*'}, 'body': json.dumps(result) }
⚠️The city name must match OpenWeatherMap's format. "New York" works but "new york" might not. Always URL-encode spaces if calling from code.

Lambda Code — All Routes Handler
chat_lambda.pyPython
import boto3, json, os db = boto3.resource('dynamodb').Table('chat_connections') APIGW = os.environ.get('APIGW_ENDPOINT', '') def lambda_handler(event, ctx): route = event['requestContext']['routeKey'] cid = event['requestContext']['connectionId'] if route == '$connect': # Save connection db.put_item(Item={'connectionId': cid}) return {'statusCode': 200} if route == '$disconnect': # Remove connection db.delete_item(Key={'connectionId': cid}) return {'statusCode': 200} if route == 'sendMessage': body = json.loads(event.get('body', '{}')) message = body.get('message', '') # Get all active connections conns = db.scan()['Items'] apigw_client = boto3.client( 'apigatewaymanagementapi', endpoint_url=APIGW ) stale = [] for conn in conns: try: apigw_client.post_to_connection( ConnectionId = conn['connectionId'], Data = json.dumps({'from': cid[:8], 'msg': message}) ) except Exception: stale.append(conn['connectionId']) # Clean up stale connections for s in stale: db.delete_item(Key={'connectionId': s}) return {'statusCode': 200} return {'statusCode': 400}
💡The APIGW_ENDPOINT must be HTTPS not WSS. Example: https://abc123.execute-api.us-east-1.amazonaws.com/prod. This is how Lambda posts messages back to connected clients.

import boto3, json, urllib.parse from datetime import datetime rek = boto3.client('rekognition') db = boto3.resource('dynamodb').Table('image_analysis') def lambda_handler(event, ctx): bucket = event['Records'][0]['s3']['bucket']['name'] key = urllib.parse.unquote_plus( event['Records'][0]['s3']['object']['key']) s3ref = {'S3Object': {'Bucket': bucket, 'Name': key}} # 1. Detect object labels labels_resp = rek.detect_labels(Image=s3ref, MaxLabels=10) labels = [{'name': l['Name'], 'confidence': round(l['Confidence'], 2)} for l in labels_resp['Labels']] # 2. Detect text in image text_resp = rek.detect_text(Image=s3ref) texts = [t['DetectedText'] for t in text_resp['TextDetections'] if t['Type'] == 'LINE'] # 3. Detect faces faces_resp = rek.detect_faces(Image=s3ref, Attributes=['DEFAULT']) face_count = len(faces_resp['FaceDetails']) # 4. Save results to DynamoDB result = { 'imageKey': key, 'analyzedAt': datetime.utcnow().isoformat(), 'bucket': bucket, 'labels': json.dumps(labels), 'texts': json.dumps(texts), 'face_count': face_count, 'top_label': labels[0]['name'] if labels else 'unknown' } db.put_item(Item=result) print(json.dumps(result, indent=2)) return result

2
Create Lambda
Python 3.12 — Policy: AmazonDynamoDBFullAccess — paste code below.
3
Create REST API with two endpoints
POST /score and GET /leaderboard — both proxy to the same Lambda. Deploy to stage prod.
4
Submit test scores
POST /score with: {"player":"Alice","game":"chess","score":2400} — submit 5 different players.
5
Query leaderboard
GET /leaderboard?game=chess — returns top 10 sorted by score.
Lambda Code
leaderboard_lambda.pyPython
import boto3, json, time from boto3.dynamodb.conditions import Key table = boto3.resource('dynamodb').Table('leaderboard') def lambda_handler(event, ctx): method = event['httpMethod'] if method == 'POST': b = json.loads(event['body']) # Upsert: keep highest score per player per game existing = table.get_item(Key={ 'player_id': b['player'], 'game_id': b['game'] }).get('Item') if not existing or b['score'] > existing['score']: table.put_item(Item={ 'player_id': b['player'], 'game_id': b['game'], 'score': b['score'], 'updated_at': int(time.time()) }) msg = 'New high score!' else: msg = 'Score not beaten' return {'statusCode': 200, 'body': json.dumps({'status': msg})} if method == 'GET': game = event.get('queryStringParameters', {}).get('game', '') # Query GSI sorted by score (highest first) resp = table.query( IndexName = 'game-score-index', KeyConditionExpression = Key('game_id').eq(game), ScanIndexForward = False, # descending Limit = 10 ) board = [{'rank': i+1, 'player': r['player_id'], 'score': int(r['score'])} for i, r in enumerate(resp['Items'])] return {'statusCode': 200, 'body': json.dumps({'game': game, 'leaderboard': board})}
⚠️DynamoDB GSI sort keys are eventually consistent by default. The leaderboard might take a second to reflect new scores. Use ScanIndexForward=False to get descending order — DynamoDB sorts by sort key (score) ascending by default.

Lambda Code
doc_processor.pyPython
import boto3, json, os, uuid, urllib.parse from datetime import datetime textract = boto3.client('textract') sns = boto3.client('sns') table = boto3.resource('dynamodb').Table('documents') SNS_ARN = os.environ['SNS_TOPIC_ARN'] def lambda_handler(event, ctx): bucket = event['Records'][0]['s3']['bucket']['name'] key = urllib.parse.unquote_plus( event['Records'][0]['s3']['object']['key']) # Extract text with Textract response = textract.detect_document_text( Document={'S3Object': {'Bucket': bucket, 'Name': key}} ) # Collect all LINE blocks lines = [b['Text'] for b in response['Blocks'] if b['BlockType'] == 'LINE'] full_text = '\n'.join(lines) word_count = len(full_text.split()) doc_id = str(uuid.uuid4()) # Save to DynamoDB (truncate text if very large) table.put_item(Item={ 'docId': doc_id, 's3Key': key, 'bucket': bucket, 'line_count': len(lines), 'word_count': word_count, 'extracted_at': datetime.utcnow().isoformat(), 'full_text': full_text[:380000] # DynamoDB item limit 400KB }) # Send notification sns.publish( TopicArn = SNS_ARN, Subject = f"Document Processed: {key}", Message = ( f"Document: {key}\n" f"Lines extracted: {len(lines)}\n" f"Word count: {word_count}\n" f"Doc ID: {doc_id}\n\n" f"First 500 chars:\n{full_text[:500]}" ) ) print(f"Processed {key}: {word_count} words, {len(lines)} lines") return {'statusCode': 200, 'docId': doc_id}

Real-Time Analytics Dashboard
Clickstream events from a website are sent to Kinesis Data Stream. A Lambda consumer processes them in real time, aggregates metrics, and stores them in DynamoDB. A dashboard API serves live stats.

Producer Lambda + Consumer Lambda + Stats Lambda
analytics_lambdas.pyPython
import boto3, json, base64, os, time # ============================================================ # PRODUCER LAMBDA — receives POST /event from API Gateway # ============================================================ kinesis = boto3.client('kinesis') STREAM = 'clickstream' def producer_handler(event, ctx): b = json.loads(event['body']) b['timestamp'] = int(time.time() * 1000) kinesis.put_record( StreamName = STREAM, Data = json.dumps(b).encode('utf-8'), PartitionKey = b.get('page', 'default') ) return {'statusCode': 200, 'body': json.dumps({'status': 'event queued'})} # ============================================================ # CONSUMER LAMBDA — triggered by Kinesis, batch of up to 100 # ============================================================ table = boto3.resource('dynamodb').Table('page_views') def consumer_handler(event, ctx): page_counts = {} for record in event['Records']: # Decode Kinesis record payload = json.loads( base64.b64decode(record['kinesis']['data']).decode('utf-8') ) page = payload.get('page', 'unknown') page_counts[page] = page_counts.get(page, 0) + 1 # Batch update DynamoDB with atomic ADD for page, count in page_counts.items(): table.update_item( Key = {'page': page}, UpdateExpression = 'ADD #v :c', ExpressionAttributeNames = {'#v': 'views'}, ExpressionAttributeValues = {':c': count} ) print(f"Processed {len(event['Records'])} events: {page_counts}") # ============================================================ # STATS LAMBDA — GET /stats, returns dashboard data # ============================================================ def stats_handler(event, ctx): items = table.scan()['Items'] # Sort by views descending sorted_pages = sorted(items, key=lambda x: int(x.get('views', 0)), reverse=True) result = [{'page': p['page'], 'views': int(p.get('views', 0))} for p in sorted_pages] return { 'statusCode': 200, 'headers': {'Access-Control-Allow-Origin': '*'}, 'body': json.dumps({'pages': result, 'total_events': sum(p['views'] for p in result)}) }
💡Note: producer_handler, consumer_handler, and stats_handler can be in separate Lambda functions (recommended) or you can use environment variables to route to the right handler in a single function. For learning, keep them separate.
