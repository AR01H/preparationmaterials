Section 01

What is Backend Engineering?

The backend is everything the user never sees — the server, business logic, databases, authentication, APIs, caching layers, message queues, and infrastructure that power the product. A backend engineer architects, builds, secures, and scales these systems.

Frontend

What users interact with — HTML, CSS, JavaScript running in the browser or a mobile app.

Backend

Servers, APIs, databases, auth systems, queues, and logic. This guide covers all of it.

Full Stack

Engineers who work confidently on both sides. Backend knowledge is the harder half.

Core Responsibilities

  • Design and expose APIs for frontends and third-party clients
  • Model and query databases efficiently
  • Implement authentication & authorization
  • Write secure code and harden services against attacks
  • Optimize for performance — latency, throughput, memory
  • Design systems that scale horizontally
  • Deploy and monitor via CI/CD pipelines

REST API Design — CRUD Example (Python)

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Dict

app = FastAPI()

class User(BaseModel):
    name: str
    email: str

db: Dict[int, User] = {}
counter = 0

@app.post("/users", status_code=201)
def create_user(user: User):
    global counter
    counter += 1
    db[counter] = user
    return {"id": counter, **user.dict()}

@app.get("/users/{user_id}")
def get_user(user_id: int):
    if user_id not in db:
        raise HTTPException(status_code=404, detail="User not found")
    return {"id": user_id, **db[user_id].dict()}

@app.put("/users/{user_id}")
def update_user(user_id: int, user: User):
    if user_id not in db:
        raise HTTPException(status_code=404, detail="User not found")
    db[user_id] = user
    return {"id": user_id, **user.dict()}

@app.delete("/users/{user_id}", status_code=204)
def delete_user(user_id: int):
    if user_id not in db:
        raise HTTPException(status_code=404, detail="User not found")
    del db[user_id]

Redis Caching Pattern (Node.js)

const redis = require('redis');
const client = redis.createClient();
await client.connect();

async function getUser(userId) {
  const cacheKey = `user:${userId}`;

  // 1. Check cache first
  const cached = await client.get(cacheKey);
  if (cached) return JSON.parse(cached);

  // 2. Miss — query DB
  const user = await db.query('SELECT * FROM users WHERE id = $1', [userId]);

  // 3. Store in cache for 5 minutes
  await client.setEx(cacheKey, 300, JSON.stringify(user.rows[0]));

  return user.rows[0];
}

JWT Authentication Flow

import jwt, bcrypt
from datetime import datetime, timedelta

SECRET = "supersecretkey"  # store in env variable!

# 1 — Hash password on registration
def hash_password(plain: str) -> str:
    return bcrypt.hashpw(plain.encode(), bcrypt.gensalt()).decode()

# 2 — Verify password on login
def verify_password(plain: str, hashed: str) -> bool:
    return bcrypt.checkpw(plain.encode(), hashed.encode())

# 3 — Issue JWT token
def create_token(user_id: int) -> str:
    payload = {
        "sub": str(user_id),
        "exp": datetime.utcnow() + timedelta(hours=24)
    }
    return jwt.encode(payload, SECRET, algorithm="HS256")

# 4 — Verify JWT on protected routes
def decode_token(token: str) -> dict:
    try:
        return jwt.decode(token, SECRET, algorithms=["HS256"])
    except jwt.ExpiredSignatureError:
        raise Exception("Token expired")
    except jwt.InvalidTokenError:
        raise Exception("Invalid token")

Role-Based Access Control (RBAC)

# Middleware pattern: check user role before route handler
from functools import wraps
from flask import request, jsonify, g

def require_role(*roles):
    def decorator(f):
        @wraps(f)
        def wrapper(*args, **kwargs):
            user = g.current_user  # set by auth middleware
            if user.role not in roles:
                return jsonify({"error": "Forbidden"}), 403
            return f(*args, **kwargs)
        return wrapper
    return decorator

# Usage

@require_role("admin", "superadmin")

    ...

Cybersecurity for Backend Engineers

Security is not a feature — it is a discipline baked into every line of code. Learn the OWASP Top 10. These are the most exploited vulnerabilities in web apps worldwide.

1. SQL Injection — Prevention

# ❌ VULNERABLE — never do this
query = f"SELECT * FROM users WHERE email = '{user_input}'"

# ✅ SAFE — use parameterized queries / prepared statements
cursor.execute("SELECT * FROM users WHERE email = %s", (user_input,))

# With SQLAlchemy ORM — automatically safe
user = session.query(User).filter(User.email == user_input).first()

2. XSS (Cross-Site Scripting) — Prevention

# Backend: always escape / sanitize output
import html

user_input = ''
safe = html.escape(user_input)
# Result: <script>alert("xss")</script>

# Set Content-Security-Policy header in all responses
from fastapi.middleware.trustedhost import TrustedHostMiddleware
from starlette.middleware.base import BaseHTTPMiddleware

class SecurityHeadersMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        response = await call_next(request)
        response.headers["Content-Security-Policy"] = "default-src 'self'"
        response.headers["X-Content-Type-Options"] = "nosniff"
        response.headers["X-Frame-Options"] = "DENY"
        return response

6. Input Validation

from pydantic import BaseModel, EmailStr, validator
import re

class RegisterRequest(BaseModel):
    username: str
    email: EmailStr
    password: str

    @validator('username')
    def username_valid(cls, v):
        if not re.match(r'^[a-zA-Z0-9_]{3,30}$', v):
            raise ValueError('Invalid username')
        return v

    @validator('password')
    def password_strong(cls, v):
        if len(v) < 8:
            raise ValueError('Password too short')
        if not re.search(r'[A-Z]', v):
            raise ValueError('Must contain uppercase')
        return v

Security Checklist

  • ✅ Use HTTPS everywhere (TLS 1.2+)
  • ✅ Parameterized queries — never string-format SQL
  • ✅ Hash passwords with bcrypt/argon2 (min cost factor 12)
  • ✅ Validate and sanitize all inputs server-side
  • ✅ Set security headers (CSP, HSTS, X-Frame-Options)
  • ✅ Rate limit authentication endpoints
  • ✅ Keep dependencies updated — use pip audit / npm audit
  • ✅ Log security events (failed logins, permission denials)
  • ✅ Never expose stack traces to clients
  • ✅ Use short-lived JWTs + refresh token rotation
OWASP Top 10 to Study: Injection, Broken Auth, Insecure Deserialization, Security Misconfiguration, Vulnerable Components, XXE, IDOR, SSRF, Logging Failures, Cryptographic Failures.

Async I/O — Don't Block the Event Loop

import asyncio
import httpx

# ❌ Sequential — waits for each request
async def fetch_sequential(urls):
    results = []
    async with httpx.AsyncClient() as client:
        for url in urls:
            r = await client.get(url)
            results.append(r.json())
    return results

# ✅ Concurrent — all requests fire simultaneously
async def fetch_concurrent(urls):
    async with httpx.AsyncClient() as client:
        tasks = [client.get(url) for url in urls]
        responses = await asyncio.gather(*tasks)
    return [r.json() for r in responses]

Pagination — Never Return Unbounded Results

# Cursor-based pagination (scalable)
@app.get("/users")
def list_users(cursor: int = 0, limit: int = 20):
    limit = min(limit, 100)  # cap max page size
    users = (
        session.query(User)
        .filter(User.id > cursor)
        .order_by(User.id)
        .limit(limit)
        .all()
    )
    next_cursor = users[-1].id if len(users) == limit else None
    return {"data": users, "next_cursor": next_cursor}

Caching Strategies

Cache-Aside

App checks cache → miss → load DB → write cache. Most flexible. You control TTL and invalidation.

Write-Through

Write to DB and cache simultaneously. Cache always fresh but adds write latency.

Write-Behind

Write to cache first, async flush to DB. Fastest writes, risk of data loss on crash.

System Design

Scalability Patterns

Vertical Scaling

Bigger machine (more CPU/RAM). Simple, but has limits and creates a single point of failure.

Horizontal Scaling

More machines behind a load balancer. Stateless services required. Standard for production.

Database Sharding

Split data across multiple DB instances by key (e.g., user_id % N). Complex but necessary at scale.

Read Replicas

Replicate DB reads to secondary nodes. Write to primary, read from replicas. Reduces load.

Microservices vs Monolith

MonolithMicroservices
Team SizeSmall–mediumLarge, multiple teams
DeploymentSimple, single unitComplex, independent

Message Queues — Decouple Services

# RabbitMQ / Kafka producer (Python — pika)
import pika, json

connection = pika.BlockingConnection(pika.ConnectionParameters('localhost'))
channel = connection.channel()
channel.queue_declare(queue='order.created', durable=True)

def publish_order_event(order: dict):
    channel.basic_publish(
        exchange='',
        routing_key='order.created',
        body=json.dumps(order),
        properties=pika.BasicProperties(delivery_mode=2)  # persistent
    )

# Consumer in a separate service
def handle_order(ch, method, props, body):
    order = json.loads(body)
    print(f"Processing order {order['id']}")
    # do work ...
    ch.basic_ack(delivery_tag=method.delivery_tag)

channel.basic_consume(queue='order.created', on_message_callback=handle_order)
channel.start_consuming()

Circuit Breaker Pattern

import time

class CircuitBreaker:
    def __init__(self, threshold=5, timeout=30):
        self.failures = 0
        self.threshold = threshold
        self.timeout = timeout
        self.opened_at = None

    def call(self, fn, *args, **kwargs):
        # If circuit is open, check timeout
        if self.opened_at:
            if time.time() - self.opened_at < self.timeout:
                raise Exception("Circuit is OPEN — service unavailable")
            else:
                self.opened_at = None  # half-open: try again
        try:
            result = fn(*args, **kwargs)
            self.failures = 0  # reset on success
            return result
        except Exception as e:
            self.failures += 1
            if self.failures >= self.threshold:
                self.opened_at = time.time()  # trip the breaker
            raise e

Repository Pattern — Decouple Business Logic from DB

from abc import ABC, abstractmethod

class UserRepository(ABC):
    @abstractmethod
    def find_by_id(self, user_id: int): ...

    @abstractmethod
    def save(self, user): ...

class PostgresUserRepository(UserRepository):
    def __init__(self, session):
        self.session = session

    def find_by_id(self, user_id: int):
        return self.session.query(User).get(user_id)

    def save(self, user):
        self.session.add(user)
        self.session.commit()

# Business logic has no idea about the DB
class UserService:
    def __init__(self, repo: UserRepository):
        self.repo = repo

    def promote_to_admin(self, user_id: int):
        user = self.repo.find_by_id(user_id)
        if not user:
            raise ValueError("User not found")
        user.role = "admin"
        self.repo.save(user)

# Simple in-process event bus
from collections import defaultdict
from typing import Callable

class EventBus:
    def __init__(self):
        self._listeners = defaultdict(list)

    def subscribe(self, event: str, handler: Callable):
        self._listeners[event].append(handler)

    def publish(self, event: str, payload: dict):
        for handler in self._listeners[event]:
            handler(payload)

Idempotency Keys — Safe Retries

# Prevent duplicate payments / operations on retry
@app.post("/payments")
async def create_payment(request: Request, payment: PaymentRequest):
    idempotency_key = request.headers.get("Idempotency-Key")

    if idempotency_key:
        # Check if we've seen this key before
        cached = await redis.get(f"idem:{idempotency_key}")
        if cached:
            return json.loads(cached)  # return cached result

    result = await process_payment(payment)

    if idempotency_key:
        # Cache result for 24 hours
        await redis.setex(f"idem:{idempotency_key}", 86400, json.dumps(result))

    return result

from fastapi import Request
import time, uuid

class RequestTracingMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        # ── BEFORE handler ──
        request_id = request.headers.get("X-Request-ID", str(uuid.uuid4()))
        start = time.perf_counter()

        # Attach to request state for downstream use
        request.state.request_id = request_id

        logger.info({
            "event":      "request_start",
            "request_id": request_id,
            "method":     request.method,
            "path":       request.url.path,
        })

        # ── CALL handler ──
        try:
            response = await call_next(request)
        except Exception as exc:
            logger.error({"event": "request_error", "error": str(exc)})
            raise

        # ── AFTER handler ──
        duration_ms = (time.perf_counter() - start) * 1000
        response.headers["X-Request-ID"]    = request_id
        response.headers["X-Response-Time"] = f"{duration_ms:.1f}ms"

        logger.info({
            "event":      "request_end",
            "request_id": request_id,
            "status":     response.status_code,
            "duration_ms": round(duration_ms, 1),
        })

        return response

app.add_middleware(RequestTracingMiddleware)

import ipaddress, socket
from urllib.parse import urlparse

BLOCKED_HOSTS = {'localhost', '0.0.0.0'}
PRIVATE_RANGES = [
    ipaddress.ip_network('10.0.0.0/8'),
    ipaddress.ip_network('172.16.0.0/12'),
    ipaddress.ip_network('192.168.0.0/16'),
    ipaddress.ip_network('127.0.0.0/8'),
    ipaddress.ip_network('169.254.0.0/16'),  # link-local
]

def is_safe_url(url: str) -> bool:
    parsed = urlparse(url)
    host = parsed.hostname

    if not host or host in BLOCKED_HOSTS:
        return False

    try:
        ip = ipaddress.ip_address(socket.gethostbyname(host))
        for private_range in PRIVATE_RANGES:
            if ip in private_range:
                return False  # block internal IPs
    except Exception:
        return False

    return True

@app.post("/fetch-url")

Multipart File Upload — FastAPI

from fastapi import FastAPI, UploadFile, File, HTTPException
import aiofiles, os, hashlib, mimetypes

app = FastAPI()

UPLOAD_DIR = "/var/app/uploads"
ALLOWED_TYPES = {"image/jpeg", "image/png", "image/webp", "application/pdf"}
MAX_SIZE = 10 * 1024 * 1024  # 10 MB

@app.post("/upload")
async def upload_file(file: UploadFile = File(...)):
    # Validate MIME type
    content_type = file.content_type
    if content_type not in ALLOWED_TYPES:
        raise HTTPException(400, f"File type {content_type} not allowed")

    # Read in chunks to avoid memory exhaustion
    size = 0
    chunks = []
    hasher = hashlib.sha256()

    async for chunk in file:
        size += len(chunk)
        if size > MAX_SIZE:
            raise HTTPException(413, "File too large")
        chunks.append(chunk)
        hasher.update(chunk)

    file_hash = hasher.hexdigest()
    ext = mimetypes.guess_extension(content_type) or ".bin"
    filename = f"{file_hash}{ext}"
    dest = os.path.join(UPLOAD_DIR, filename)

    async with aiofiles.open(dest, "wb") as f:
        for chunk in chunks:
            await f.write(chunk)

    return {"filename": filename, "size": size, "sha256": file_hash}

Kafka & Event Streaming

Kafka is a distributed, persistent, high-throughput event log. Unlike RabbitMQ (which deletes messages after consumption), Kafka retains messages for a configurable period — any consumer can replay from the beginning.

Core Concepts

Topic

Named stream of records. Think of it as a database table for events. Partitioned for parallelism.

Partition

Ordered, immutable log within a topic. Each partition can be consumed independently by one consumer per group.

Consumer Group

Multiple consumers sharing the work of reading a topic. Each partition goes to exactly one consumer in the group.

Offset

Position of a message in a partition. Consumers commit offsets to track progress. Replay by resetting offset.

Producer — Python (kafka-python)

# 1. Sliding Window — rate limiting, moving average
from collections import deque
import time

class SlidingWindowRateLimiter:
    def __init__(self, max_requests: int, window_seconds: int):
        self.max = max_requests
        self.window = window_seconds
        self.timestamps: dict[str, deque] = {}

    def is_allowed(self, client_id: str) -> bool:
        now = time.time()
        if client_id not in self.timestamps:
            self.timestamps[client_id] = deque()

        dq = self.timestamps[client_id]
        cutoff = now - self.window

        while dq and dq[0] < cutoff:   # remove expired
            dq.popleft()

        if len(dq) >= self.max:
            return False               # over limit

        dq.append(now)
        return True

# 2. LRU Cache — evict least recently used (O(1) get + put)
from functools import lru_cache
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity: int):
        self.cap = capacity
        self.cache = OrderedDict()

    def get(self, key):
        if key not in self.cache:
            return -1
        self.cache.move_to_end(key)  # mark as recently used
        return self.cache[key]

    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.cap:
            self.cache.popitem(last=False)  # evict oldest

# ── Pattern: Rate Limiter with Token Bucket ──
class TokenBucket:
    def __init__(self, capacity: int, refill_rate: float):
        """capacity: max tokens; refill_rate: tokens per second"""
        self.capacity = capacity
        self.tokens = capacity
        self.refill_rate = refill_rate
        self.last_refill = time.monotonic()

    def consume(self, tokens: int = 1) -> bool:
        now = time.monotonic()
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity, self.tokens + elapsed * self.refill_rate)
        self.last_refill = now

        if self.tokens >= tokens:
            self.tokens -= tokens
            return True   # allowed
        return False      # denied

# ── Pattern: Consistent Hashing ──
import hashlib, bisect

class ConsistentHash:
    def __init__(self, nodes: list[str], replicas: int = 100):
        self.replicas = replicas
        self.ring: dict[int, str] = {}
        self.sorted_keys: list[int] = []
        for node in nodes:
            self.add_node(node)

    def add_node(self, node: str):
        for i in range(self.replicas):
            key = self._hash(f"{node}:{i}")
            self.ring[key] = node
            bisect.insort(self.sorted_keys, key)

    def get_node(self, key: str) -> str:
        h = self._hash(key)
        idx = bisect.bisect_right(self.sorted_keys, h) % len(self.sorted_keys)
        return self.ring[self.sorted_keys[idx]]

    def _hash(self, key: str) -> int:
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

import uuid

def acquire_lock(redis_client, lock_name: str, ttl: int = 30) -> str | None:
    lock_id = str(uuid.uuid4())
    acquired = redis_client.set(
        f"lock:{lock_name}", lock_id,
        nx=True,        # only set if not exists
        ex=ttl,         # auto-expire
    )
    return lock_id if acquired else None

import hmac, hashlib

# Every major provider (Stripe, GitHub, Twilio) signs webhook payloads
# Always verify the signature before processing!

def verify_stripe_webhook(payload: bytes, sig_header: str, secret: str) -> bool:
    """Stripe sends: Stripe-Signature: t=timestamp,v1=hash"""
    parts = dict(item.split("=", 1) for item in sig_header.split(","))
    timestamp = parts.get("t")
    signature = parts.get("v1")

    signed_payload = f"{timestamp}.{payload.decode()}"
    expected = hmac.new(
        secret.encode(), signed_payload.encode(), hashlib.sha256
    ).hexdigest()

    # Constant-time comparison to prevent timing attacks
    if not hmac.compare_digest(expected, signature):
        raise ValueError("Invalid webhook signature")

    # Reject replays older than 5 minutes
    if abs(time.time() - int(timestamp)) > 300:
        raise ValueError("Webhook timestamp too old")

    return True

# Reliable webhook delivery with retries and exponential backoff
import httpx, asyncio

RETRY_DELAYS = [10, 30, 60, 300, 1800]  # seconds: 10s, 30s, 1m, 5m, 30m

async def deliver_webhook(endpoint_url: str, payload: dict, attempt: int = 0):
    try:
        async with httpx.AsyncClient(timeout=10) as client:
            response = await client.post(
                endpoint_url,
                json=payload,
                headers={
                    "Content-Type": "application/json",
                    "X-Webhook-ID": payload["webhook_id"],
                    "X-Signature":  sign_payload(payload),
                }
            )
            if response.status_code < 300:
                await log_delivery_success(payload["webhook_id"])
                return

            # 4xx — don't retry (endpoint misconfigured)
            if 400 <= response.status_code < 500:
                await log_delivery_failed(payload["webhook_id"], "4xx error", final=True)
                return

    except (httpx.TimeoutException, httpx.ConnectError) as e:
        pass  # fall through to retry logic

    # Retry with exponential backoff
    if attempt < len(RETRY_DELAYS):
        delay = RETRY_DELAYS[attempt]
        await asyncio.sleep(delay)
        await deliver_webhook(endpoint_url, payload, attempt + 1)
    else:
        await log_delivery_failed(payload["webhook_id"], "max retries", final=True)
