Module 01

Introduction to Node.js

What it is, how it works, and why it matters

What is Node.js?

Node.js is a JavaScript runtime built on Chrome's V8 JavaScript engine. It allows you to run JavaScript outside of the browser — on servers, in CLI tools, and desktop applications. Released in 2009 by Ryan Dahl, Node.js revolutionized backend development by enabling JavaScript everywhere.

⚡
Non-blocking I/O

Operations don't block the thread. Node handles thousands of concurrent connections efficiently.

🔄
Event-Driven

Everything is driven by events. The event loop processes callbacks when I/O completes.

📦
npm Ecosystem

Access to 2M+ packages. The largest open-source software registry in the world.

🔁
Single-threaded

One main thread, but async I/O means it never sits idle waiting for disk or network.

The Node.js Architecture

Understanding how Node works under the hood is essential. Node.js uses the Event Loop — a single-threaded mechanism that offloads expensive operations to the OS or thread pool, then picks up results via callbacks.

The Event Loop — Simplified

When Node.js starts, it initializes the event loop, processes the input script (which may make async API calls, schedule timers, or call process.nextTick()), then begins processing the event loop.

Event Loop Phases
timers → setTimeout, setInterval callbacks
pending I/O → I/O callbacks deferred to next loop
idle/prepare→ internal use only
poll → retrieve new I/O events; execute callbacks
check → setImmediate() callbacks
close cbs → socket.on('close', ...) etc.

When to use Node.js

Node.js excels at I/O-heavy applications: REST APIs, real-time chat, streaming services, microservices. It's not ideal for CPU-intensive tasks like video encoding or complex math — though Worker Threads address this (Module 12).

Module 03

File System (fs)

Reading, writing, and managing files and directories

The fs module provides an API for interacting with the file system. Every method has both synchronous (blocking) and asynchronous (non-blocking) variants.

Reading Files

JavaScript fs-read.js 
const fs = require('fs');
const fsPromises = require('fs/promises'); // promise-based API

// 1. Synchronous — blocks until complete (avoid in production)
try {
  const data = fs.readFileSync('./data.txt', 'utf8');
  console.log(data);
} catch (err) {
  console.error('File not found');
}

// 2. Async with callback
fs.readFile('./data.txt', 'utf8', (err, data) => {
  if (err) { console.error(err); return; }
  console.log(data);
});

// 3. Promise-based (recommended)
async function readConfig() {
  const content = await fsPromises.readFile('./config.json', 'utf8');
  return JSON.parse(content);
}

Writing Files

JavaScript fs-write.js 
const { writeFile, appendFile, mkdir, rm, rename } = require('fs/promises');

async function demoFileOps() {
  // Write (creates or overwrites)
  await writeFile('./output.txt', 'Hello World\n');

  // Append to file
  await appendFile('./output.txt', 'Second line\n');

  // Write JSON
  const user = { name: 'Alice', age: 30, role: 'admin' };
  await writeFile('./user.json', JSON.stringify(user, null, 2));

  // Create directory (recursive handles nested dirs)
  await mkdir('./logs/2024', { recursive: true });

  // Rename / move file
  await rename('./output.txt', './logs/output.txt');

  // Delete file
  await rm('./temp.txt', { force: true }); // force = don't throw if missing
}

demoFileOps();

Directory Operations

JavaScript fs-dir.js 
const fs = require('fs/promises');
const path = require('path');

async function listFiles(dir) {
  const entries = await fs.readdir(dir, { withFileTypes: true });

  for (const entry of entries) {
    const fullPath = path.join(dir, entry.name);
    if (entry.isDirectory()) {
      console.log(`📁 ${entry.name}/`);
      await listFiles(fullPath); // recurse
    } else {
      const stats = await fs.stat(fullPath);
      console.log(`  📄 ${entry.name} (${stats.size} bytes)`);
    }
  }
}

listFiles('.');

Module 04

Events & EventEmitter

The heart of Node's event-driven architecture

Node.js is fundamentally event-driven. The EventEmitter class is the backbone of most Node.js core APIs — http.Server, fs.ReadStream, and many more all extend EventEmitter.

Basic EventEmitter Usage

JavaScript events-basic.js 
const { EventEmitter } = require('events');

const emitter = new EventEmitter();

// Register a listener
emitter.on('greet', (name) => {
  console.log(`Hello, ${name}!`);
});

// Listen only once
emitter.once('connect', () => {
  console.log('Connected! (fires once only)');
});

// Emit events
emitter.emit('greet', 'Alice'); // Hello, Alice!
emitter.emit('greet', 'Bob');   // Hello, Bob!
emitter.emit('connect');        // fires
emitter.emit('connect');        // ignored (once)

// Remove a specific listener
const handler = () => console.log('done');
emitter.on('done', handler);
emitter.off('done', handler);    // remove it

Extending EventEmitter (Custom Classes)

JavaScript logger.js 
const { EventEmitter } = require('events');

class Logger extends EventEmitter {
  log(message, level = 'info') {
    const entry = { message, level, timestamp: new Date() };
    console.log(`[${level.toUpperCase()}] ${message}`);
    this.emit('log', entry);

    if (level === 'error') {
      this.emit('error-log', entry);
    }
  }
}

const logger = new Logger();

// Listen for all logs
logger.on('log', ({ message, level }) => {
  // Could write to file, send to monitoring etc.
  console.log(`Persisting: ${level} - ${message}`);
});

// Listen for errors specifically
logger.on('error-log', (entry) => {
  // Send alert, notify Slack, etc.
  console.error('🚨 Error logged!', entry);
});

logger.log('Server started');
logger.log('Database connection failed', 'error');
⚠️
Memory Leak Warning

By default, Node.js warns if more than 10 listeners are added for a single event. This often indicates a listener leak (e.g. adding listeners inside loops). Use emitter.setMaxListeners(n) if you legitimately need more, or emitter.removeAllListeners('event') to clean up.

Module 05

Async: Callbacks, Promises & Async/Await

Mastering asynchronous JavaScript in Node

Asynchronous programming is the core of Node.js. Understanding the three patterns — callbacks, promises, and async/await — is essential for every Node developer.

Callbacks (The Old Way)

JavaScript callbacks.js 
const fs = require('fs');

// Node.js error-first callback convention:
// First arg is error, second is result
fs.readFile('./data.json', 'utf8', function(err, data) {
  if (err) {
    console.error('Error:', err.message);
    return; // always return on error!
  }
  console.log(JSON.parse(data));
});

// ❌ Callback Hell — deeply nested, hard to read
readUser(1, (err, user) => {
  if (err) return handleErr(err);
  readPosts(user.id, (err, posts) => {
    if (err) return handleErr(err);
    renderPage(user, posts, (err, html) => {
      if (err) return handleErr(err);
      sendResponse(html); // deeply nested!
    });
  });
});

Promises

JavaScript promises.js 
// Creating a Promise
function fetchUser(id) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (id > 0) {
        resolve({ id, name: 'Alice', email: 'alice@example.com' });
      } else {
        reject(new Error('Invalid user ID'));
      }
    }, 100);
  });
}

// Using .then() / .catch() / .finally()
fetchUser(1)
  .then(user => console.log('User:', user.name))
  .catch(err => console.error('Error:', err.message))
  .finally(() => console.log('Done'));

// Promise.all — run in parallel, wait for all
Promise.all([fetchUser(1), fetchUser(2), fetchUser(3)])
  .then(([u1, u2, u3]) => console.log(u1, u2, u3));

// Promise.allSettled — run all, get all results
Promise.allSettled([fetchUser(1), fetchUser(-1)])
  .then(results => {
    results.forEach(r => {
      if (r.status === 'fulfilled') console.log(r.value);
      else console.error(r.reason.message);
    });
  });

Async/Await (Modern Way)

JavaScript async-await.js 
// async/await makes async code look synchronous
async function loadUserProfile(userId) {
  try {
    // Sequential — each waits for the previous
    const user = await fetchUser(userId);
    const posts = await fetchPosts(user.id);

    // Parallel — both fire at same time
    const [followers, following] = await Promise.all([
      fetchFollowers(userId),
      fetchFollowing(userId)
    ]);

    return { user, posts, followers, following };
  } catch (err) {
    console.error('Failed to load profile:', err.message);
    throw err; // re-throw to let caller handle
  }
}

// Using the async function
loadUserProfile(42)
  .then(profile => console.log(profile))
  .catch(console.error);

// Top-level await (ESM or Node 16+)
// const profile = await loadUserProfile(42);
✅
Best Practice

Use async/await for most cases. Use Promise.all when operations are independent and can run in parallel. Always handle errors with try/catch.

🚫
Common Mistake

Don't use await in a loop when promises can run in parallel. Use Promise.all(array.map(...)) instead for better performance.

Module 06

Streams & Buffers

Efficient data processing without loading everything into memory

Streams are one of Node.js's most powerful features. Instead of loading an entire file or response into memory, streams process data in chunks — perfect for large files, video, and real-time data.

Types of Streams

📖
Readable

Source of data. Examples: fs.createReadStream(), HTTP request body, stdin.

✍️
Writable

Destination for data. Examples: fs.createWriteStream(), HTTP response, stdout.

🔄
Duplex

Both readable and writable. Example: TCP sockets, WebSocket connections.

🔧
Transform

Modify data as it passes through. Examples: zlib.createGzip(), crypto streams.

Reading and Writing with Streams

JavaScript streams.js 
const fs = require('fs');
const zlib = require('zlib');
const { pipeline } = require('stream/promises');

// ✅ Pipe large file — memory efficient
async function compressFile(input, output) {
  await pipeline(
    fs.createReadStream(input),
    zlib.createGzip(),          // transform: compress
    fs.createWriteStream(output)
  );
  console.log('File compressed!');
}

compressFile('./bigfile.csv', './bigfile.csv.gz');

// Read a stream chunk by chunk
async function processLines(filename) {
  const readStream = fs.createReadStream(filename, { encoding: 'utf8' });
  let lineBuffer = '';

  for await (const chunk of readStream) {
    lineBuffer += chunk;
    const lines = lineBuffer.split('\n');
    lineBuffer = lines.pop(); // keep incomplete line
    for (const line of lines) {
      console.log('Line:', line);
    }
  }
}

Buffers

A Buffer is a fixed-size chunk of memory outside the V8 heap, used to work with binary data directly. Streams produce Buffers by default.

JavaScript buffers.js 
// Create buffers
const buf1 = Buffer.from('Hello Node.js');           // from string
const buf2 = Buffer.alloc(10);                        // 10 zeros
const buf3 = Buffer.from([0x48, 0x65, 0x6c, 0x6c, 0x6f]); // from bytes

// Convert buffer to string
console.log(buf1.toString('utf8'));   // 'Hello Node.js'
console.log(buf1.toString('base64')); // base64 encoded
console.log(buf1.toString('hex'));    // hex encoded

// Buffer operations
console.log(buf1.length);              // 13 (bytes)
console.log(buf1[0]);                  // 72 (ASCII for 'H')

// Combine buffers
const combined = Buffer.concat([buf1, buf3]);
console.log(combined.toString());     // 'Hello Node.jsHello'

HTTP & Building Servers

Creating servers with Node's built-in http module

Creating an HTTP Server

JavaScript server.js 
const http = require('http');
const url = require('url');

const server = http.createServer((req, res) => {
  const parsedUrl = new URL(req.url, `http://${req.headers.host}`);
  const pathname = parsedUrl.pathname;
  const method = req.method;

  // Set response headers
  res.setHeader('Content-Type', 'application/json');
  res.setHeader('X-Powered-By', 'Node.js');

  // Simple router
  if (pathname === '/' && method === 'GET') {
    res.writeHead(200);
    res.end(JSON.stringify({ message: 'Welcome to the API' }));

  } else if (pathname === '/users' && method === 'GET') {
    res.writeHead(200);
    res.end(JSON.stringify([{ id: 1, name: 'Alice' }]));

  } else if (method === 'POST') {
    // Read request body
    let body = '';
    req.on('data', chunk => { body += chunk; });
    req.on('end', () => {
      const data = JSON.parse(body);
      res.writeHead(201);
      res.end(JSON.stringify({ created: data }));
    });

  } else {
    res.writeHead(404);
    res.end(JSON.stringify({ error: 'Not Found' }));
  }
});

server.listen(3000, () => {
  console.log('Server running at http://localhost:3000');
});

Module 08

Express.js Framework

The most popular Node.js web framework

Express.js is a minimal, fast, and flexible web framework for Node.js. It provides routing, middleware support, and HTTP utilities that make building APIs clean and maintainable.

Express Fundamentals

JavaScript app.js 
const express = require('express');
const app = express();

// Built-in middleware
app.use(express.json());              // parse JSON bodies
app.use(express.urlencoded({ extended: true })); // parse form data

// GET route
app.get('/', (req, res) => {
  res.json({ message: 'Hello from Express!' });
});

// Route parameters
app.get('/users/:id', (req, res) => {
  const { id } = req.params;
  res.json({ userId: id });
});

// Query strings: GET /search?q=nodejs&page=2
app.get('/search', (req, res) => {
  const { q, page = 1 } = req.query;
  res.json({ query: q, page: Number(page) });
});

// POST with body
app.post('/users', (req, res) => {
  const { name, email } = req.body;
  if (!name || !email) {
    return res.status(400).json({ error: 'Name and email required' });
  }
  // ... save to DB
  res.status(201).json({ id: Date.now(), name, email });
});

app.listen(3000, () => console.log('Express server on port 3000'));

Middleware

Middleware are functions with access to (req, res, next). They form a pipeline — each middleware can process the request, modify it, or terminate it. Call next() to pass to the next middleware.

JavaScript middleware.js 
// 1. Application-level middleware (runs on every request)
app.use((req, res, next) => {
  req.requestTime = new Date();
  console.log(`${req.method} ${req.path} - ${req.requestTime.toISOString()}`);
  next(); // MUST call next() or request hangs!
});

// 2. Route-level middleware (auth guard)
function requireAuth(req, res, next) {
  const token = req.headers['authorization']?.split(' ')[1];
  if (!token) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  req.user = verifyToken(token); // decode JWT etc.
  next();
}

// Apply middleware to specific routes
app.get('/profile', requireAuth, (req, res) => {
  res.json(req.user);
});

// 3. Error-handling middleware (4 params!)
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(err.status || 500).json({
    error: err.message || 'Internal Server Error'
  });
});

MongoDB with Mongoose

Mongoose provides an elegant MongoDB object modeling for Node.js with schema validation, middleware (hooks), and query building.

JavaScript models/User.js 
const mongoose = require('mongoose');
const bcrypt = require('bcryptjs');

const userSchema = new mongoose.Schema({
  name:  { type: String, required: true, trim: true },
  email: { type: String, required: true, unique: true, lowercase: true },
  password: { type: String, required: true, minlength: 8 },
  role: { type: String, enum: ['user', 'admin'], default: 'user' },
  createdAt: { type: Date, default: Date.now }
}, { timestamps: true }); // auto createdAt + updatedAt

// Pre-save hook — hash password before saving
userSchema.pre('save', async function(next) {
  if (!this.isModified('password')) return next();
  this.password = await bcrypt.hash(this.password, 12);
  next();
});

// Instance method
userSchema.methods.comparePassword = async function(candidate) {
  return bcrypt.compare(candidate, this.password);
};

// Static method
userSchema.statics.findByEmail = function(email) {
  return this.findOne({ email: email.toLowerCase() });
};

module.exports = mongoose.model('User', userSchema);

PostgreSQL with the pg Library

JavaScript db/postgres.js 
const { Pool } = require('pg');

// Use a connection pool (not individual clients)
const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  max: 20,          // max pool size
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 2000
});

// Query with parameterized values (prevents SQL injection!)
async function getUser(email) {
  const result = await pool.query(
    'SELECT * FROM users WHERE email = $1',
    [email]  // parameterized — NEVER use string interpolation!
  );
  return result.rows[0];
}

// Transaction example
async function transferFunds(fromId, toId, amount) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    await client.query('UPDATE accounts SET balance = balance - $1 WHERE id = $2', [amount, fromId]);
    await client.query('UPDATE accounts SET balance = balance + $1 WHERE id = $2', [amount, toId]);
    await client.query('COMMIT');
    console.log('Transfer successful');
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release(); // always release!
  }
}

Authentication & JWT

Secure user authentication with JSON Web Tokens

JWT Authentication Flow

How it works: User logs in → server validates credentials → server creates a signed JWT containing user data → client stores JWT → client sends JWT in every subsequent request → server verifies JWT signature and extracts user data.

JavaScript auth/jwt.js 
const jwt = require('jsonwebtoken');
const bcrypt = require('bcryptjs');
const User = require('../models/User');

const JWT_SECRET = process.env.JWT_SECRET;
const JWT_EXPIRES = '7d';

// Generate JWT token
function generateToken(userId) {
  return jwt.sign(
    { userId, iat: Math.floor(Date.now() / 1000) },
    JWT_SECRET,
    { expiresIn: JWT_EXPIRES }
  );
}

// POST /auth/register
exports.register = async (req, res, next) => {
  try {
    const { name, email, password } = req.body;
    const existing = await User.findOne({ email });
    if (existing) return res.status(409).json({ error: 'Email already registered' });

    const user = await User.create({ name, email, password }); // password hashed in pre-save hook
    const token = generateToken(user._id);

    res.status(201).json({ token, user: { id: user._id, name, email } });
  } catch (err) { next(err); }
};

// POST /auth/login
exports.login = async (req, res, next) => {
  try {
    const { email, password } = req.body;
    const user = await User.findOne({ email }).select('+password');
    if (!user || !(await bcrypt.compare(password, user.password))) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }
    const token = generateToken(user._id);
    res.json({ token });
  } catch (err) { next(err); }
};

// Auth middleware — protects routes
exports.protect = async (req, res, next) => {
  try {
    const authHeader = req.headers.authorization;
    if (!authHeader?.startsWith('Bearer ')) {
      return res.status(401).json({ error: 'No token provided' });
    }
    const token = authHeader.split(' ')[1];
    const decoded = jwt.verify(token, JWT_SECRET); // throws if invalid/expired
    req.user = await User.findById(decoded.userId);
    next();
  } catch (err) {
    res.status(401).json({ error: 'Invalid or expired token' });
  }
};
🔐
Security: JWT Best Practices

Use a strong random secret (32+ characters). Store tokens in httpOnly cookies for web apps (not localStorage — vulnerable to XSS). Set short expiration times (15 min access + 7 day refresh). Always use HTTPS. Never store sensitive data in the JWT payload (it's base64 encoded, not encrypted).

Worker Threads

True parallelism for CPU-intensive tasks

By default, Node.js runs on a single thread. For CPU-bound operations like image processing, encryption, or complex calculations, Worker Threads let you run JavaScript in parallel on separate threads.

JavaScript workers/main.js 
const { Worker, isMainThread, parentPort, workerData } = require('worker_threads');

if (isMainThread) {
  // This is the main thread
  function runWorker(data) {
    return new Promise((resolve, reject) => {
      const worker = new Worker(__filename, { workerData: data });
      worker.on('message', resolve);
      worker.on('error', reject);
      worker.on('exit', (code) => {
        if (code !== 0) reject(new Error(`Worker exited with code ${code}`));
      });
    });
  }

  // Run heavy computation in parallel workers
  async function main() {
    console.log('Main thread: starting workers...');
    const [r1, r2] = await Promise.all([
      runWorker({ start: 1, end: 1_000_000 }),
      runWorker({ start: 1_000_001, end: 2_000_000 })
    ]);
    console.log('Results:', r1 + r2);
  }

  main();
} else {
  // This is a worker thread
  const { start, end } = workerData;
  let sum = 0;
  for (let i = start; i <= end; i++) sum += i;
  parentPort.postMessage(sum);
}

Clustering & Child Processes

Scaling Node.js to utilize all CPU cores

The cluster module allows you to create child processes (workers) that share the same server port, each running on a separate CPU core. This is how you scale a Node.js app to take advantage of multi-core machines.

JavaScript cluster.js 
const cluster = require('cluster');
const os = require('os');
const http = require('http');

const numCPUs = os.cpus().length;

if (cluster.isPrimary) {
  console.log(`Primary ${process.pid} running on ${numCPUs} CPUs`);

  // Fork workers for each CPU
  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }

  // Auto-restart crashed workers
  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died (code: ${code}). Restarting...`);
    cluster.fork(); // replace the dead worker
  });

} else {
  // Workers share the TCP connection
  http.createServer((req, res) => {
    res.writeHead(200);
    res.end(`Handled by worker ${process.pid}`);
  }).listen(3000);

  console.log(`Worker ${process.pid} started`);
}

WebSockets with Socket.IO

Real-time bidirectional communication

WebSockets enable real-time, full-duplex communication between client and server over a persistent connection. Socket.IO adds reliability, namespaces, rooms, and automatic fallbacks.

const http = require('http');
const { Server } = require('socket.io');
const express = require('express');

const app = express();

const io = new Server(server, {
  cors: { origin: '*', methods: ['GET', 'POST'] }
});

// Middleware for authentication
io.use((socket, next) => {
  const token = socket.handshake.auth.token;
  try {
    socket.user = verifyJWT(token);
    next();
  } catch {
    next(new Error('Authentication error'));
  }
});

io.on('connection', (socket) => {
  console.log(`User ${socket.user.name} connected`);

  // Join a room
  socket.on('join-room', (roomId) => {
    socket.join(roomId);
    socket.to(roomId).emit('user-joined', socket.user.name);
    console.log(`${socket.user.name} joined room ${roomId}`);
  });

  // Chat message
  socket.on('chat-message', ({ roomId, message }) => {
    const msg = {
      text: message,
      sender: socket.user.name,
      timestamp: new Date()
    };
    io.to(roomId).emit('chat-message', msg); // broadcast to room
  });

  // Typing indicator
  socket.on('typing', (roomId) => {
    socket.to(roomId).emit('user-typing', socket.user.name);
  });

  // Disconnect
  socket.on('disconnect', () => {
    console.log(`${socket.user.name} disconnected`);
  });
});

server.listen(3000);

Emit Cheatsheet

socket.emit('event', data) → send to this client only
socket.to(room).emit(...) → send to room, exclude sender
io.to(room).emit(...) → send to everyone in room including sender
io.emit('event', data) → broadcast to ALL connected clients

Custom Error Classes

JavaScript errors/AppError.js 
class AppError extends Error {
  constructor(message, statusCode, code) {
    super(message);
    this.statusCode = statusCode;
    this.code = code;
    this.isOperational = true; // vs programming errors
    Error.captureStackTrace(this, this.constructor);
  }
}

class NotFoundError extends AppError {
  constructor(resource) {
    super(`${resource} not found`, 404, 'NOT_FOUND');
  }
}

class ValidationError extends AppError {
  constructor(message, fields) {
    super(message, 422, 'VALIDATION_ERROR');
    this.fields = fields;
  }
}

// Global error handler middleware
function errorHandler(err, req, res, next) {
  if (err.isOperational) {
    return res.status(err.statusCode).json({
      status: 'error',
      code: err.code,
      message: err.message,
      ...(err.fields && { fields: err.fields })
    });
  }

  // Programming error — don't leak details
  console.error('PROGRAMMING ERROR:', err);
  res.status(500).json({
    status: 'error',
    message: 'Something went wrong'
  });
}

module.exports = { AppError, NotFoundError, ValidationError, errorHandler };

Caching with Redis

JavaScript cache/redis.js 
const { createClient } = require('redis');

const client = createClient({ url: process.env.REDIS_URL });
await client.connect();

// Cache middleware
function cache(ttlSeconds = 60) {
  return async (req, res, next) => {
    const key = `cache:${req.originalUrl}`;
    try {
      const cached = await client.get(key);
      if (cached) {
        res.setHeader('X-Cache', 'HIT');
        return res.json(JSON.parse(cached));
      }
      // Intercept res.json to cache the response
      const originalJson = res.json.bind(res);
      res.json = async (data) => {
        await client.setEx(key, ttlSeconds, JSON.stringify(data));
        res.setHeader('X-Cache', 'MISS');
        return originalJson(data);
      };
      next();
    } catch (err) {
      next(); // cache failure should not break the app
    }
  };
}

// Usage: cache expensive endpoints
app.get('/api/products', cache(300), productController.getAll);
