SH
.sh Shell Script Files — Linux & Git Bash
Create reusable scripts that run on Linux, macOS, and Git Bash on Windows
A .sh file is just a text file with shell commands. Save it as myscript.sh, make it executable with chmod +x myscript.sh, then run it with ./myscript.sh or bash myscript.sh.
myscript.sh — structure & basics
#!/bin/bash
# ↑ ALWAYS first line — tells system which shell to use
# Lines starting with # are comments

# Run this file:
# chmod +x myscript.sh   ← make it executable (once)
# ./myscript.sh           ← run it
# bash myscript.sh        ← also works

echo "Starting script..."

# Exit with code 0 = success, non-zero = error
exit 0
variables.sh — variables, args, input
#!/bin/bash

# ── Set variables (NO spaces around =) ──
NAME="Alice"
AGE=30

# ── Read variable with $ ──
echo "Hello, $NAME! You are $AGE years old."

# ── Command line arguments ──
# ./script.sh arg1 arg2
echo "Script name: $0"
echo "First arg:   $1"
echo "Second arg:  $2"
echo "All args:    $@"
echo "Arg count:   $#"

# ── Read user input ──
read -p "Enter your city: " CITY
echo "You are from $CITY"

# ── Capture command output into variable ──
CURRENT_DATE=$(date)
echo "Today is: $CURRENT_DATE"

# ── String operations ──
FULL="Hello World"
echo ${#FULL}         # length = 11
echo ${FULL:0:5}      # substring = "Hello"
echo ${FULL/World/Bash} # replace = "Hello Bash"
conditions.sh
#!/bin/bash

NUM=10

# ── Number comparisons ──
# -eq -ne -gt -lt -ge -le
if [ $NUM -gt 5 ]; then
  echo "Greater than 5"
elif [ $NUM -eq 5 ]; then
  echo "Equal to 5"
else
  echo "Less than 5"
fi

# ── String comparisons ──
STR="hello"
if [ "$STR" = "hello" ]; then
  echo "Strings match"
fi

# ── File / folder checks ──
if [ -f "file.txt" ]; then   # -f = file exists
  echo "File exists"
fi
if [ -d "myfolder" ]; then  # -d = directory exists
  echo "Folder exists"
fi
if [ -z "$VAR" ]; then       # -z = string is empty
  echo "VAR is empty"
fi

# ── AND / OR conditions ──
if [ $NUM -gt 5 ] && [ $NUM -lt 20 ]; then
  echo "Between 5 and 20"
fi

# ── Case statement ──
DAY="Monday"
case $DAY in
  Monday|Tuesday|Wednesday|Thursday|Friday)
    echo "Weekday" ;;
  Saturday|Sunday)
    echo "Weekend" ;;
  *) echo "Unknown" ;;
esac
loops.sh
#!/bin/bash

# ── For loop over list ──
for FRUIT in apple banana mango; do
  echo "Fruit: $FRUIT"
done

# ── For loop with range ──
for i in {1..5}; do
  echo "Count: $i"
done

# ── For loop C-style ──
for (( i=0; i<5; i++ )); do
  echo "i = $i"
done

# ── Loop over files ──
for FILE in *.txt; do
  echo "Processing: $FILE"
done

# ── While loop ──
COUNT=1
while [ $COUNT -le 3 ]; do
  echo "Loop $COUNT"
  (( COUNT++ ))
done

# ── Until loop (opposite of while) ──
N=0
until [ $N -ge 3 ]; do
  echo "N=$N"
  (( N++ ))
done

# ── break & continue ──
for i in {1..10}; do
  [ $i -eq 5 ] && break    # stop at 5
  [ $i -eq 3 ] && continue # skip 3
  echo $i
done
functions.sh
#!/bin/bash

# ── Define a function ──
greet() {
  echo "Hello, $1!"   # $1 = first arg passed to function
}

# ── Call the function ──
greet "Alice"
greet "Bob"

# ── Function with return value ──
# (use echo + capture, not return — return is exit code only)
add() {
  echo $(( $1 + $2 ))
}
RESULT=$(add 5 3)
echo "5 + 3 = $RESULT"

# ── Function with error handling ──
check_file() {
  if [ ! -f "$1" ]; then
    echo "ERROR: File '$1' not found"
    return 1   # non-zero = failure
  fi
  echo "File '$1' found ✅"
  return 0
}
check_file "myfile.txt"

# ── Exit on any error (best practice) ──
set -e           # script stops if any command fails
set -u           # error if unset variable used
set -o pipefail  # catch errors in pipes
deploy.sh — real-world deployment script
#!/bin/bash
# Real-world example: project build + deploy script
set -e

PROJECT_DIR="/var/www/myapp"
BACKUP_DIR="/var/backups/myapp"
BRANCH="main"
LOG_FILE="deploy.log"

# ── Logging function ──
log() {
  echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a $LOG_FILE
}

log "Starting deployment..."

# ── Check we're in right directory ──
if [ ! -d "$PROJECT_DIR" ]; then
  log "ERROR: Project dir not found"
  exit 1
fi

cd $PROJECT_DIR

# ── Backup current version ──
log "Backing up..."
mkdir -p $BACKUP_DIR
cp -r . $BACKUP_DIR/backup_$(date +%Y%m%d_%H%M%S)

# ── Pull latest code ──
log "Pulling from $BRANCH..."
git pull origin $BRANCH

# ── Install dependencies ──
log "Installing dependencies..."
npm install --production

# ── Build ──
log "Building..."
npm run build

# ── Restart service ──
log "Restarting service..."
pm2 restart myapp || pm2 start dist/index.js --name myapp

log "Deployment complete ✅"
On Windows with Git Bash: .sh scripts run fine in Git Bash terminal. Line endings must be Unix (LF), not Windows (CRLF). In VS Code: bottom-right corner → click "CRLF" → change to "LF" before saving your .sh file.
BAT
.bat Batch Files — Windows CMD
Windows-native automation scripts, double-click to run
A .bat file is a text file with Windows CMD commands. Save it as script.bat and double-click it — or run from CMD. Use @echo off at the top to hide command echo. Works on all Windows versions with no installation needed.
myscript.bat — structure & basics
@echo off
:: ↑ ALWAYS first line — hides command echo
:: Lines starting with :: or REM are comments
REM This is also a comment

:: Pause at end so window doesn't close immediately
echo Starting script...
echo Hello from Windows!

:: Change to script's own directory
cd /D "%~dp0"

:: Set window title
title My Script

:: Exit with code 0 = success
exit /B 0

:: At end, add pause so console stays open:
:: pause
