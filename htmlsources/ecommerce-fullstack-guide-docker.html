01

System Architecture

Overview

This app follows a classic 3-tier architecture: a React frontend, a RESTful Node.js API, and a PostgreSQL database — plus Redis for caching/sessions and Nginx as a reverse proxy.

🖥️

Frontend (React)

React + Vite SPA. Talks to the API via REST. Deployed to S3/CloudFront or as a Docker container.

⚙️

Backend (Node.js)

Express REST API with JWT auth, Stripe payments, and business logic. Runs in Docker.

🗄️

Database (PostgreSQL)

Primary data store: users, products, orders, reviews. Runs on AWS RDS in production.

⚡

Cache (Redis)

Session storage, cart caching, and product catalog caching. Reduces DB load significantly.

🔀

Nginx (Reverse Proxy)

Routes traffic to frontend and backend. Handles SSL termination in production.

☁️

AWS (Cloud)

EC2 for servers, RDS for Postgres, ElastiCache for Redis, S3 + CloudFront for frontend assets.

🏗️ Architecture Flow

User → Nginx → React App / Node API → PostgreSQL + Redis
In production on AWS: User → CloudFront → ALB → EC2 → RDS + ElastiCache

const db    = require('../config/db');
const redis = require('../config/redis');

// GET /api/products?page=1&limit=12&category=shoes
exports.getProducts = async (req, res) => {
  try {
    const { page = 1, limit = 12, category } = req.query;
    const cacheKey = `products:${page}:${limit}:${category||'all'}`;

    // Try cache first
    const cached = await redis.get(cacheKey);
    if (cached) return res.json(JSON.parse(cached));

    const offset = (page - 1) * limit;
    let query = `SELECT * FROM products WHERE is_active = true`;
    const params = [];

    if (category) {
      params.push(category);
      query += ` AND category = $${params.length}`;
    }
    params.push(limit, offset);
    query += ` ORDER BY created_at DESC LIMIT $${params.length-1} OFFSET $${params.length}`;

    const { rows } = await db.query(query, params);

    // Cache for 5 minutes
    await redis.setex(cacheKey, 300, JSON.stringify(rows));
    res.json(rows);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
};

PostgreSQL Schema

Database
backend/src/models/schema.sqlsql
-- Enable UUID extension
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- USERS
CREATE TABLE users (
  id         UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  email      VARCHAR(255) UNIQUE NOT NULL,
  password   VARCHAR(255) NOT NULL,
  name       VARCHAR(255) NOT NULL,
  role       VARCHAR(20) DEFAULT 'customer' CHECK (role IN ('customer','admin')),
  avatar     TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- CATEGORIES
CREATE TABLE categories (
  id   SERIAL PRIMARY KEY,
  name VARCHAR(100) UNIQUE NOT NULL,
  slug VARCHAR(100) UNIQUE NOT NULL
);

-- PRODUCTS
CREATE TABLE products (
  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name        VARCHAR(255) NOT NULL,
  description TEXT,
  price       NUMERIC(10,2) NOT NULL,
  stock       INTEGER NOT NULL DEFAULT 0,
  category_id INTEGER REFERENCES categories(id),
  images      TEXT[],          -- array of image URLs
  is_active   BOOLEAN DEFAULT true,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- ORDERS
CREATE TABLE orders (
  id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id         UUID REFERENCES users(id) ON DELETE SET NULL,
  status          VARCHAR(30) DEFAULT 'pending'
    CHECK (status IN ('pending','paid','shipped','delivered','cancelled')),
  total           NUMERIC(10,2) NOT NULL,
  stripe_pi_id    VARCHAR(255),  -- Stripe PaymentIntent ID
  shipping_addr   JSONB,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- ORDER ITEMS
CREATE TABLE order_items (
  id         SERIAL PRIMARY KEY,
  order_id   UUID REFERENCES orders(id) ON DELETE CASCADE,
  product_id UUID REFERENCES products(id) ON DELETE SET NULL,
  quantity   INTEGER NOT NULL,
  price      NUMERIC(10,2) NOT NULL  -- price at time of purchase
);

-- REVIEWS
CREATE TABLE reviews (
  id         SERIAL PRIMARY KEY,
  product_id UUID REFERENCES products(id) ON DELETE CASCADE,
  user_id    UUID REFERENCES users(id) ON DELETE CASCADE,
  rating     SMALLINT CHECK (rating BETWEEN 1 AND 5),
  comment    TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(product_id, user_id)  -- one review per user per product
);

-- INDEXES for performance
CREATE INDEX idx_products_category ON products(category_id);
CREATE INDEX idx_orders_user       ON orders(user_id);
CREATE INDEX idx_order_items_order ON order_items(order_id);

Cart Controller (using Redis)

backend/src/controllers/cart.controller.jsjs
const redis = require('../config/redis');

const cartKey = (userId) => `cart:${userId}`;

// GET /api/cart
exports.getCart = async (req, res) => {
  const data = await redis.get(cartKey(req.user.id));
  res.json(data ? JSON.parse(data) : { items: [] });
};

// POST /api/cart/add  { productId, quantity }
exports.addToCart = async (req, res) => {
  const { productId, quantity = 1 } = req.body;
  const key  = cartKey(req.user.id);
  const data = await redis.get(key);
  const cart = data ? JSON.parse(data) : { items: [] };

  const existing = cart.items.find(i => i.productId === productId);
  if (existing) existing.quantity += quantity;
  else cart.items.push({ productId, quantity });

  // Expire cart after 7 days
  await redis.setex(key, 604800, JSON.stringify(cart));
  res.json(cart);
};

// DELETE /api/cart/clear
exports.clearCart = async (req, res) => {
  await redis.del(cartKey(req.user.id));
  res.json({ success: true });
};

JWT Authentication

Security
backend/src/controllers/auth.controller.jsjs
const bcrypt = require('bcryptjs');
const jwt    = require('jsonwebtoken');
const db     = require('../config/db');

// POST /api/auth/register
exports.register = async (req, res) => {
  try {
    const { email, password, name } = req.body;
    const hash = await bcrypt.hash(password, 12);

    const { rows } = await db.query(
      `INSERT INTO users (email, password, name)
       VALUES ($1, $2, $3) RETURNING id, email, name, role`,
      [email, hash, name]
    );

    const token = jwt.sign(
      { id: rows[0].id, role: rows[0].role },
      process.env.JWT_SECRET,
      { expiresIn: '7d' }
    );
    res.status(201).json({ token, user: rows[0] });
  } catch (err) {
    if (err.code === '23505')  // unique violation
      return res.status(409).json({ error: 'Email already in use' });
    res.status(500).json({ error: err.message });
  }
};

// POST /api/auth/login
exports.login = async (req, res) => {
  const { email, password } = req.body;
  const { rows } = await db.query(
    `SELECT * FROM users WHERE email = $1`, [email]
  );
  if (!rows[0] || !await bcrypt.compare(password, rows[0].password))
    return res.status(401).json({ error: 'Invalid credentials' });

  const token = jwt.sign(
    { id: rows[0].id, role: rows[0].role },
    process.env.JWT_SECRET,
    { expiresIn: '7d' }
  );
  const { password: _, ...user } = rows[0];  // strip password
  res.json({ token, user });
};
backend/src/middleware/auth.middleware.jsjs
const jwt = require('jsonwebtoken');

exports.authenticate = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    res.status(401).json({ error: 'Invalid token' });
  }
};

exports.isAdmin = (req, res, next) => {
  if (req.user.role !== 'admin')
    return res.status(403).json({ error: 'Forbidden' });
  next();
};

Stripe Payments

Payments
backend/src/controllers/order.controller.jsjs
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);
const db     = require('../config/db');
const redis  = require('../config/redis');

// POST /api/orders/checkout
exports.createOrder = async (req, res) => {
  const client = await db.connect();
  try {
    await client.query('BEGIN');

    const { shippingAddress } = req.body;
    const cartData = await redis.get(`cart:${req.user.id}`);
    if (!cartData) throw new Error('Cart is empty');
    const cart = JSON.parse(cartData);

    // Fetch prices from DB (never trust client prices)
    let total = 0;
    const enrichedItems = [];
    for (const item of cart.items) {
      const { rows } = await client.query(
        'SELECT id, price, stock FROM products WHERE id = $1',
        [item.productId]
      );
      if (!rows[0] || rows[0].stock < item.quantity)
        throw new Error(`Insufficient stock for product ${item.productId}`);
      total += rows[0].price * item.quantity;
      enrichedItems.push({ ...item, price: rows[0].price });
    }

    // Create Stripe PaymentIntent
    const paymentIntent = await stripe.paymentIntents.create({
      amount: Math.round(total * 100),  // cents
      currency: 'usd',
      metadata: { userId: req.user.id },
    });

    // Insert order
    const { rows: [order] } = await client.query(
      `INSERT INTO orders (user_id, total, stripe_pi_id, shipping_addr)
       VALUES ($1, $2, $3, $4) RETURNING *`,
      [req.user.id, total, paymentIntent.id, JSON.stringify(shippingAddress)]
    );

    // Insert order items + decrement stock
    for (const item of enrichedItems) {
      await client.query(
        `INSERT INTO order_items (order_id, product_id, quantity, price)
         VALUES ($1, $2, $3, $4)`,
        [order.id, item.productId, item.quantity, item.price]
      );
      await client.query(
        `UPDATE products SET stock = stock - $1 WHERE id = $2`,
        [item.quantity, item.productId]
      );
    }

    await client.query('COMMIT');
    await redis.del(`cart:${req.user.id}`);  // clear cart

    res.json({
      orderId: order.id,
      clientSecret: paymentIntent.client_secret  // send to frontend
    });
  } catch (err) {
    await client.query('ROLLBACK');
    res.status(400).json({ error: err.message });
  } finally {
    client.release();
  }
};
💡 Key Practice Point

Notice the database transaction (BEGIN/COMMIT/ROLLBACK). This ensures that if stock decrementation fails, the order also rolls back — preventing overselling.

import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import api from '../api/axios';

export const useStore = create(
  persist(
    (set, get) => ({
      // AUTH
      user:  null,
      token: null,
      login: async (email, password) => {
        const { data } = await api.post('/auth/login', { email, password });
        set({ user: data.user, token: data.token });
      },
      logout: () => set({ user: null, token: null, cart: [] }),

      // CART (local state, synced with Redis via API)
      cart: [],
      addToCart: async (productId, quantity = 1) => {
        const { data } = await api.post('/cart/add', { productId, quantity });
        set({ cart: data.items });
      },
      fetchCart: async () => {
        const { data } = await api.get('/cart');
        set({ cart: data.items });
      },
    }),
    { name: 'ecommerce-store', partialize: s => ({ token: s.token, user: s.user }) }
  )
);

Checkout with Stripe

frontend/src/pages/CheckoutPage.jsxjsx
import { useState } from 'react';
import { useStripe, useElements, CardElement } from '@stripe/react-stripe-js';
import api from '../api/axios';

export default function CheckoutPage() {
  const stripe   = useStripe();
  const elements = useElements();
  const [loading, setLoading] = useState(false);
  const [error, setError]     = useState(null);

  const handleSubmit = async (e) => {
    e.preventDefault();
    setLoading(true);

    // 1. Create order + get clientSecret
    const { data } = await api.post('/orders/checkout', {
      shippingAddress: { street: '...', city: '...' }
    });

    // 2. Confirm payment with Stripe
    const { error: stripeError } = await stripe.confirmCardPayment(
      data.clientSecret,
      { payment_method: { card: elements.getElement(CardElement) } }
    );

    if (stripeError) { setError(stripeError.message); setLoading(false); return; }
    // 3. Redirect to success page
    window.location.href = `/order/${data.orderId}/success`;
  };

  return (
    <form onSubmit={handleSubmit}>
      <CardElement />
      {error && <p style={{color:'red'}}>{error}</p>}
      <button type="submit" disabled={!stripe || loading}>
        {loading ? 'Processing...' : 'Pay Now'}
      </button>
    </form>
  );
}

AWS Deployment

Production

This is the production architecture. We use EC2 for the app, RDS for PostgreSQL, ElastiCache for Redis, and S3+CloudFront for the frontend.

Step-by-Step AWS Setup

  • 1
    Create a VPC with public + private subnets

    EC2 and ALB go in public subnets. RDS and ElastiCache go in private subnets. This is the standard secure architecture.

  • 2
    Launch RDS PostgreSQL

    Go to RDS → Create database → PostgreSQL 16 → Free tier (db.t3.micro) → Set credentials → Place in private subnet. Note the endpoint URL.

  • 3
    Launch ElastiCache Redis

    Go to ElastiCache → Create cluster → Redis OSS → t3.micro → Place in private subnet. Note the endpoint URL.

  • 4
    Create ECR Repositories & Push Docker Images

    Create two repos: ecommerce/backend and ecommerce/frontend. Build and push images using the commands below.

  • 5
    Launch EC2 (Ubuntu 22.04, t3.small+)

    Install Docker, pull your images from ECR, and run with docker-compose.prod.yml pointing to RDS/ElastiCache endpoints.

  • 6
    Create S3 Bucket + CloudFront for Frontend

    Upload the React build to S3, enable static website hosting, create a CloudFront distribution pointing at S3. This is fast and cheap.

  • 7
    Configure Security Groups

    EC2 allows 80/443 from internet. RDS allows 5432 only from EC2 security group. ElastiCache allows 6379 only from EC2 security group.

  • 8
    SSL Certificate with ACM + ALB

    Request a free SSL cert from ACM, create an ALB with HTTPS listener pointing to your EC2 instance. This handles HTTPS termination.

        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Install & Test Backend
        working-directory: ./backend
        run: |
          npm ci
          npm test

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id:     ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region:            ap-south-1

      - name: Login to ECR
        uses: aws-actions/amazon-ecr-login@v2

      - name: Build & Push Docker images
        run: |
          docker build -t ${{ secrets.ECR_URI }}/ecommerce/backend:latest ./backend
          docker push ${{ secrets.ECR_URI }}/ecommerce/backend:latest

      - name: Deploy to EC2 via SSH
        uses: appleboy/ssh-action@v1
        with:
          host:     ${{ secrets.EC2_HOST }}
          username: ubuntu
          key:      ${{ secrets.EC2_SSH_KEY }}
          script: |
            aws ecr get-login-password --region ap-south-1 \
              | docker login --username AWS --password-stdin ${{ secrets.ECR_URI }}
            docker compose -f docker-compose.prod.yml pull
            docker compose -f docker-compose.prod.yml up -d
            docker image prune -f

⚠️ Security

Never commit .env files to git. Add them to .gitignore. For production on AWS, use AWS Secrets Manager or Parameter Store to manage secrets — never hardcode them in docker-compose files.

What to Practice — Learning Path

  • 1
    Run locally with Docker Compose

    Get all 5 containers running together. Test API endpoints with Postman or curl. Practice docker compose logs and debugging.

  • 2
    Add a feature end-to-end

    Try adding product search — DB query → controller → route → frontend API call → React component. Feel the full stack.

  • 3
    Break things intentionally

    Stop Redis — see how the cart fails gracefully. Stop Postgres — observe error handling. This builds intuition for resilient systems.

  • 4
    Deploy to AWS (Free Tier)

    Set up RDS + ElastiCache + EC2. Push your Docker images to ECR. Deploy with docker-compose.prod.yml. Get your app live on the internet.

  • 5
    Set up CI/CD pipeline

    Connect GitHub Actions to your AWS account. Push to main → tests run → Docker image builds → auto-deploys to EC2. Real DevOps workflow.
