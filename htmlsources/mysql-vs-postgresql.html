Connecting, listing databases, describing tables.

MySQL

psql -U postgres

Create table — auto-increment
MySQL
CREATE TABLE users (
  id    INT AUTO_INCREMENT PRIMARY KEY,
  name  VARCHAR(100) NOT NULL,
  email VARCHAR(255) UNIQUE,
  bio   TEXT,
  age   TINYINT UNSIGNED,
  score DECIMAL(5,2),
  created_at TIMESTAMP
    DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8mb4;
PostgreSQL
CREATE TABLE users (
  id    SERIAL PRIMARY KEY,  -- or BIGSERIAL
  name  VARCHAR(100) NOT NULL,
  email VARCHAR(255) UNIQUE,
  bio   TEXT,
  age   SMALLINT,
  score NUMERIC(5,2),
  created_at TIMESTAMPTZ
    DEFAULT NOW()
);
-- Modern alternative (SQL standard):
-- id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY

CRUD Operations

INSERT, SELECT, UPDATE, DELETE — and PostgreSQL's powerful RETURNING clause.

INSERT
MySQL
INSERT INTO users (name, email)
VALUES ('Ali', 'ali@x.com');

-- Multi-row
INSERT INTO users (name, email) VALUES
  ('Priya', 'p@x.com'),
  ('Reza',  'r@x.com');
PostgreSQL
INSERT INTO users (name, email)
VALUES ('Ali', 'ali@x.com')
RETURNING id, created_at;

-- Multi-row + return all
INSERT INTO users (name, email) VALUES
  ('Priya', 'p@x.com'),
  ('Reza',  'r@x.com')
RETURNING *;
SELECT with filtering
Both (same syntax)
SELECT id, name, email
FROM users
WHERE age BETWEEN 18 AND 35
  AND email LIKE '%@gmail.com'
  AND status IN ('active', 'pending')
ORDER BY name ASC, created_at DESC
LIMIT 20 OFFSET 40;
UPDATE & DELETE with RETURNING
MySQL
UPDATE users
SET name = 'New Name', age = 30
WHERE id = 1;
-- Need separate SELECT to see result
SELECT * FROM users WHERE id = 1;
PostgreSQL
UPDATE users
SET name = 'New Name', age = 30
WHERE id = 1
RETURNING *;  -- returns updated rows

DELETE FROM users
WHERE status = 'inactive'
RETURNING id, email;
UPSERT (insert or update on conflict)
MySQL
INSERT INTO users (id, name, score)
VALUES (1, 'Ali', 95)
ON DUPLICATE KEY UPDATE
  name  = VALUES(name),
  score = VALUES(score);
PostgreSQL
INSERT INTO users (id, name, score)
VALUES (1, 'Ali', 95)
ON CONFLICT (id)
DO UPDATE SET
  name  = EXCLUDED.name,
  score = EXCLUDED.score
RETURNING *;

-- Skip on conflict:
INSERT INTO users (email)
VALUES ('a@x.com')
ON CONFLICT DO NOTHING;

JOINs — All Types

INNER, LEFT, RIGHT, FULL, CROSS, SELF — syntax and use cases.

INNER JOIN — only matching rows
Both (same syntax)
SELECT
  u.id,
  u.name,
  o.id      AS order_id,
  o.total,
  o.created_at
FROM users u
INNER JOIN orders o ON o.user_id = u.id
WHERE o.total > 100
ORDER BY o.created_at DESC;
Returns only users who have at least one matching order. Users with no orders are excluded.
LEFT JOIN — all left rows + nulls for unmatched right
Both (same syntax)
SELECT
  u.name,
  COUNT(o.id)    AS order_count,
  COALESCE(SUM(o.total), 0) AS total_spent
FROM users u
LEFT JOIN orders o ON o.user_id = u.id
GROUP BY u.id, u.name
ORDER BY total_spent DESC;
All users appear even if they have no orders (order_count = 0, total_spent = 0).
FULL OUTER JOIN — all rows from both sides
MySQL — emulated
-- MySQL has no FULL OUTER JOIN.
-- Emulate with UNION:
SELECT u.name, o.total
FROM users u
LEFT JOIN orders o
  ON o.user_id = u.id
UNION
SELECT u.name, o.total
FROM users u
RIGHT JOIN orders o
  ON o.user_id = u.id;
PostgreSQL — native
SELECT
  u.name,
  o.total
FROM users u
FULL OUTER JOIN orders o
  ON o.user_id = u.id;
-- Rows appear from both sides.
-- Unmatched sides show NULL.
CROSS JOIN — cartesian product
Both
-- Every combination of rows (n × m rows)
SELECT s.size, c.color
FROM sizes s
CROSS JOIN colors c;
SELF JOIN — join table to itself
Both
-- Find employees and their managers
SELECT
  e.name  AS employee,
  m.name  AS manager
FROM employees e
LEFT JOIN employees m
  ON m.id = e.manager_id
ORDER BY m.name NULLS LAST;
Multiple JOINs
Both
SELECT
  u.name,
  o.id        AS order_id,
  p.name      AS product,
  oi.quantity,
  oi.unit_price
FROM users u
JOIN orders      o  ON o.user_id    = u.id
JOIN order_items oi ON oi.order_id  = o.id
JOIN products    p  ON p.id         = oi.product_id
WHERE o.created_at >= '2024-01-01'
ORDER BY o.id, p.name;

Subqueries

Scalar, correlated, IN/EXISTS/ANY/ALL — with PostgreSQL extras.

Scalar subquery — single value
Both
-- Show each order with user's total order count
SELECT
  o.id,
  o.total,
  (SELECT COUNT(*) FROM orders
   WHERE user_id = o.user_id) AS user_order_count
FROM orders o;
Subquery in FROM (derived table)
Both
SELECT dept, avg_salary
FROM (
  SELECT
    department   AS dept,
    AVG(salary)  AS avg_salary
  FROM employees
  GROUP BY department
) AS dept_avg
WHERE avg_salary > 60000;
Correlated subquery — references outer row
Both
-- Employees earning more than their dept average
SELECT name, salary, department
FROM employees e
WHERE salary > (
  SELECT AVG(salary)
  FROM employees
  WHERE department = e.department  -- ← outer ref
);
Re-executed for every outer row — can be slow on large tables. Often better rewritten as a window function or CTE.
IN / NOT IN
Both
-- Users who have placed at least one order
SELECT * FROM users
WHERE id IN (SELECT user_id FROM orders);

-- Users with NO orders at all
SELECT * FROM users
WHERE id NOT IN (
  SELECT user_id FROM orders
  WHERE user_id IS NOT NULL  -- ⚠ NULLs in NOT IN = no rows!
);
EXISTS / NOT EXISTS — often faster than IN
Both
-- Users with at least one high-value order
SELECT u.name
FROM users u
WHERE EXISTS (
  SELECT 1 FROM orders o
  WHERE o.user_id = u.id
    AND o.total > 1000
);

-- Users with NO orders
SELECT u.name
FROM users u
WHERE NOT EXISTS (
  SELECT 1 FROM orders o
  WHERE o.user_id = u.id
);
EXISTS stops at the first match (short-circuits). Prefer EXISTS over IN for large subquery results.
ANY / ALL comparisons
Both
-- Salary > any single salary in IT dept
SELECT name, salary FROM employees
WHERE salary > ANY (
  SELECT salary FROM employees
  WHERE department = 'IT'
);

-- Salary > ALL salaries in IT dept (= highest)
SELECT name, salary FROM employees
WHERE salary > ALL (
  SELECT salary FROM employees
  WHERE department = 'IT'
);
Subquery in SELECT list (column subquery)
Both
SELECT
  p.name,
  p.price,
  (
    SELECT AVG(price)
    FROM products
    WHERE category_id = p.category_id
  ) AS category_avg_price,
  p.price - (
    SELECT AVG(price)
    FROM products
    WHERE category_id = p.category_id
  ) AS diff_from_avg
FROM products p;

Aggregates + GROUP BY

COUNT, SUM, AVG, MIN/MAX, GROUP BY, HAVING, ROLLUP.

Basic aggregates
Both
SELECT
  department,
  COUNT(*)           AS headcount,
  COUNT(DISTINCT mgr) AS unique_managers,
  AVG(salary)        AS avg_salary,
  SUM(salary)        AS total_payroll,
  MIN(salary)        AS min_salary,
  MAX(salary)        AS max_salary
FROM employees
GROUP BY department
HAVING COUNT(*) > 5
ORDER BY total_payroll DESC;
ROLLUP — subtotals and grand total
Both
SELECT
  year,
  quarter,
  SUM(revenue) AS total
FROM sales
GROUP BY ROLLUP(year, quarter);
-- Produces subtotal per year + grand total
GROUP_CONCAT vs STRING_AGG — aggregate strings
MySQL
SELECT
  user_id,
  GROUP_CONCAT(
    tag ORDER BY tag
    SEPARATOR ', '
  ) AS all_tags
FROM post_tags
GROUP BY user_id;
PostgreSQL
SELECT
  user_id,
  STRING_AGG(
    tag, ', '
    ORDER BY tag
  ) AS all_tags,
  ARRAY_AGG(tag ORDER BY tag)
    AS tags_array
FROM post_tags
GROUP BY user_id;
PG PostgreSQL also has ARRAY_AGG to aggregate into a native array, and JSON_AGG / JSONB_AGG to aggregate into JSON.
FILTER clause on aggregates (PostgreSQL only)
PostgreSQL
-- Conditional aggregation without CASE WHEN
SELECT
  department,
  COUNT(*) AS total,
  COUNT(*) FILTER (WHERE status = 'active')   AS active_count,
  AVG(salary) FILTER (WHERE hired_year = 2024) AS avg_new_hire_salary
FROM employees
GROUP BY department;
The FILTER clause is cleaner than CASE WHEN inside SUM/COUNT and works with any aggregate function including window functions.

Indexes

B-Tree, composite, partial, covering, GIN, GiST — creation and use.

Basic index creation
MySQL
-- Basic index
CREATE INDEX idx_email
  ON users (email);

-- Composite index
CREATE INDEX idx_dept_salary
  ON employees (department, salary);

-- Unique index
CREATE UNIQUE INDEX idx_uq_slug
  ON posts (slug);

-- Full-text index
CREATE FULLTEXT INDEX idx_ft_body
  ON posts (title, body);
PostgreSQL
-- Basic index
CREATE INDEX idx_email
  ON users (email);

-- Composite index
CREATE INDEX idx_dept_salary
  ON employees (department, salary);

-- Partial index (only active users)
CREATE INDEX idx_active_email
  ON users (email)
  WHERE status = 'active';

-- GIN index for full-text
CREATE INDEX idx_ft_body
  ON posts USING GIN
  (to_tsvector('english', title || body));

-- Concurrent (no table lock)
CREATE INDEX CONCURRENTLY idx_name
  ON users (name);

Transactions + Locks

ACID transactions, savepoints, isolation levels, advisory locks.

Basic transaction
Both
BEGIN;
  UPDATE accounts SET balance = balance - 100 WHERE id = 1;
  UPDATE accounts SET balance = balance + 100 WHERE id = 2;
  -- Check condition
  SELECT balance FROM accounts WHERE id = 1;
COMMIT;  -- or ROLLBACK;
SAVEPOINT — partial rollback
Both
BEGIN;
  INSERT INTO orders (user_id, total) VALUES (1, 200);
  SAVEPOINT after_order;

  INSERT INTO payments (order_id, amount) VALUES (99, 200);
  -- If payment fails:
  ROLLBACK TO after_order;   -- undo only the payment
  -- Order insert is still pending

  RELEASE SAVEPOINT after_order;
COMMIT;
Isolation levels
MySQL
SET SESSION
  TRANSACTION ISOLATION LEVEL
  REPEATABLE READ; -- default

-- Levels: READ UNCOMMITTED
--         READ COMMITTED
--         REPEATABLE READ (default)
--         SERIALIZABLE
PostgreSQL
BEGIN
  ISOLATION LEVEL
  READ COMMITTED; -- default

-- Levels: READ UNCOMMITTED
--         READ COMMITTED (default)
--         REPEATABLE READ
--         SERIALIZABLE

-- Session default:
SET default_transaction_isolation
  = 'serializable';
Advisory locks (PostgreSQL only)
PostgreSQL
-- Application-level mutex (useful for cron jobs)
SELECT pg_advisory_lock(12345);
-- ... do exclusive work ...
SELECT pg_advisory_unlock(12345);

-- Non-blocking try:
SELECT pg_try_advisory_lock(12345);
-- Returns true if acquired, false if busy
Advisory locks are entirely application-managed — they don't lock any table rows, just a key. Useful for distributed cron jobs or ensuring only one process runs a migration.

Window Functions

RANK, ROW_NUMBER, LAG/LEAD, NTILE, running totals.

Ranking functions
Both (MySQL 8+)
SELECT
  name,
  department,
  salary,
  RANK()        OVER (PARTITION BY department ORDER BY salary DESC) AS rank_in_dept,
  DENSE_RANK()  OVER (PARTITION BY department ORDER BY salary DESC) AS dense_rank,
  ROW_NUMBER()  OVER (PARTITION BY department ORDER BY salary DESC) AS row_num,
  NTILE(4)      OVER (ORDER BY salary DESC)                          AS quartile
FROM employees;
RANK skips numbers after ties (1,1,3). DENSE_RANK does not skip (1,1,2). ROW_NUMBER is always unique.
LAG / LEAD — access adjacent rows
Both (MySQL 8+)
SELECT
  month,
  revenue,
  LAG(revenue, 1, 0) OVER (ORDER BY month) AS prev_month,
  LEAD(revenue, 1)   OVER (ORDER BY month) AS next_month,
  revenue - LAG(revenue, 1, revenue)
    OVER (ORDER BY month)               AS mom_change
FROM monthly_revenue;
Running totals + moving averages
Both (MySQL 8+)
SELECT
  sale_date,
  amount,
  SUM(amount) OVER (
    ORDER BY sale_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_total,
  AVG(amount) OVER (
    ORDER BY sale_date
    ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
  ) AS moving_avg_7day
FROM sales;
FIRST_VALUE / LAST_VALUE
Both (MySQL 8+)
SELECT
  name,
  department,
  salary,
  FIRST_VALUE(name) OVER (
    PARTITION BY department
    ORDER BY salary DESC
    ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
  ) AS top_earner_in_dept
FROM employees;

CTEs + Recursive Queries

WITH clause, chained CTEs, recursive trees/graphs.

Basic CTE
Both (MySQL 8+)
WITH top_customers AS (
  SELECT user_id, SUM(total) AS spent
  FROM orders
  GROUP BY user_id
  HAVING SUM(total) > 1000
)
SELECT u.name, tc.spent
FROM users u
JOIN top_customers tc ON tc.user_id = u.id
ORDER BY tc.spent DESC;
Chained CTEs
Both (MySQL 8+)
WITH
  order_totals AS (
    SELECT user_id, SUM(total) AS total
    FROM orders
    GROUP BY user_id
  ),
  ranked_customers AS (
    SELECT
      user_id,
      total,
      RANK() OVER (ORDER BY total DESC) AS rnk
    FROM order_totals
  )
SELECT u.name, rc.total, rc.rnk
FROM users u
JOIN ranked_customers rc ON rc.user_id = u.id
WHERE rc.rnk <= 10;
Recursive CTE — walk a tree / org chart
Both (MySQL 8+)
WITH RECURSIVE org_tree AS (
  -- Anchor: start from CEO (no manager)
  SELECT id, name, manager_id, 0 AS depth
  FROM employees
  WHERE manager_id IS NULL

  UNION ALL

  -- Recursive: join children to parents
  SELECT e.id, e.name, e.manager_id, ot.depth + 1
  FROM employees e
  JOIN org_tree ot ON ot.id = e.manager_id
)
SELECT
  REPEAT('  ', depth) || name AS org_chart,
  depth
FROM org_tree
ORDER BY depth, name;
Works for any tree structure: categories, comments, file systems, org charts, bill-of-materials.
CTE materialization hint (PostgreSQL)
PostgreSQL
-- Force CTE to execute once and cache result
WITH expensive_query AS MATERIALIZED (
  SELECT * FROM huge_table WHERE expensive_condition
)
SELECT * FROM expensive_query WHERE x = 1
UNION ALL
SELECT * FROM expensive_query WHERE y = 2;

-- Force inline (let planner optimize through)
WITH inline_cte AS NOT MATERIALIZED (
  SELECT * FROM users
)
SELECT * FROM inline_cte WHERE id = 1;

JSON / JSONB

PostgreSQL's powerful binary JSON — operators, indexing, querying.

JSONB vs JSON: JSONB is stored as binary (decomposed), supports GIN indexing, and query operators. JSON is stored as text (preserves whitespace/key order). Use JSONB almost always.
Create JSONB column + GIN index
PostgreSQL
CREATE TABLE products (
  id    SERIAL PRIMARY KEY,
  name  TEXT,
  attrs JSONB
);

-- GIN index enables fast @>, ?, @? etc.
CREATE INDEX idx_attrs_gin
  ON products USING GIN (attrs);

INSERT INTO products (name, attrs) VALUES
  ('Shirt',   '{"color":"red","size":42,"tags":["sale","new"]}'),
  ('Jeans',   '{"color":"blue","size":32,"tags":["denim"]}'),
  ('Sneakers','{"color":"white","size":10,"brand":"Nike"}');
JSON operators
PostgreSQL
-- -> returns JSON,  ->> returns text
SELECT attrs->'color'          -- "red"  (JSON)
SELECT attrs->>'color'         -- red   (text, no quotes)
SELECT attrs#>'{tags,0}'      -- path: first tag (JSON)
SELECT attrs#>>'{tags,0}'     -- path: first tag (text)

-- Containment: does attrs @> this JSON?
SELECT * FROM products
WHERE attrs @> '{"color":"red"}';

-- Key exists?
SELECT * FROM products WHERE attrs ? 'brand';

-- Any of these keys exist?
SELECT * FROM products WHERE attrs ?| ARRAY['brand','sku'];

-- Cast and compare
SELECT * FROM products
WHERE (attrs->>'size')::int > 10;
Update JSONB values
PostgreSQL
-- Set a key
UPDATE products
SET attrs = attrs || '{"on_sale":true}'
WHERE id = 1;

-- Remove a key
UPDATE products
SET attrs = attrs - 'on_sale'
WHERE id = 1;

-- Deep set with jsonb_set
UPDATE products
SET attrs = jsonb_set(attrs, '{nested,price}', '99.99')
WHERE id = 1;
jsonb_each — expand JSONB into rows
PostgreSQL
-- Expand each key-value pair
SELECT id, key, value
FROM products,
     jsonb_each(attrs);

-- Expand array into rows
SELECT id, tag
FROM products,
     jsonb_array_elements_text(attrs->'tags') AS tag;

Arrays + Ranges

Native arrays, unnesting, range types — PostgreSQL only.

Array column basics
PostgreSQL
CREATE TABLE articles (
  id       SERIAL PRIMARY KEY,
  title    TEXT,
  tags     TEXT[],
  scores   INT[],
  matrix   INTEGER[][]   -- 2D array
);

INSERT INTO articles (title, tags, scores) VALUES
  ('PostgreSQL', '{postgres,sql,db}', '{95,87,92}'),
  ('Arrays',     ARRAY['array','postgres'], ARRAY[80,75]);

-- Index access (1-based!)
SELECT tags[1] FROM articles;  -- 'postgres'

-- Array contains element
SELECT * FROM articles
WHERE 'postgres' = ANY(tags);

-- Array contains all of these
SELECT * FROM articles
WHERE tags @> ARRAY['sql','postgres'];

-- GIN index for fast array queries
CREATE INDEX ON articles USING GIN (tags);
UNNEST — expand array into rows
PostgreSQL
-- Expand tags into individual rows
SELECT id, unnest(tags) AS tag
FROM articles;

-- Count tag frequency across all articles
SELECT tag, COUNT(*) AS freq
FROM articles, unnest(tags) AS tag
GROUP BY tag
ORDER BY freq DESC;
Range types
PostgreSQL
CREATE TABLE bookings (
  id       SERIAL PRIMARY KEY,
  room_id  INT,
  period   TSRANGE   -- timestamp range
);

-- Exclude overlapping bookings (constraint)
ALTER TABLE bookings
  ADD CONSTRAINT no_overlap
  EXCLUDE USING GIST (
    room_id WITH =,
    period  WITH &&
  );

-- Insert a booking
INSERT INTO bookings (room_id, period) VALUES
  (1, '[2024-12-01 14:00, 2024-12-01 16:00)');

-- Find bookings overlapping a range
SELECT * FROM bookings
WHERE period &&
  '[2024-12-01 15:00, 2024-12-01 17:00)';TSRANGE
Range types: INT4RANGE, INT8RANGE, NUMRANGE, TSRANGE, TSTZRANGE, DATERANGE. The EXCLUDE constraint enforces no-overlap at the database level.

Full-Text Search

tsvector, tsquery, ranking, and highlighting.

Basic full-text search
MySQL
-- Needs FULLTEXT index first
ALTER TABLE posts
  ADD FULLTEXT (title, body);

-- Natural language search
SELECT *, MATCH(title, body)
  AGAINST('postgresql database')
  AS score
FROM posts
WHERE MATCH(title, body)
  AGAINST('postgresql database')
ORDER BY score DESC;
PostgreSQL
-- Search with ranking
SELECT
  title,
  ts_rank(
    to_tsvector('english', title||' '||body),
    to_tsquery('english', 'postgres & database')
  ) AS rank
FROM posts
WHERE
  to_tsvector('english', title||' '||body)
  @@
  to_tsquery('english', 'postgres & database')
ORDER BY rank DESC;
Stored tsvector + GIN index (fast search)
PostgreSQL
-- Add a tsvector column for performance
ALTER TABLE posts
  ADD COLUMN search_vec TSVECTOR
  GENERATED ALWAYS AS (
    to_tsvector('english', title || ' ' || body)
  ) STORED;

CREATE INDEX idx_search_vec
  ON posts USING GIN (search_vec);

-- Fast search uses index
SELECT title FROM posts
WHERE search_vec @@ websearch_to_tsquery('postgres arrays')
ORDER BY ts_rank(search_vec, websearch_to_tsquery('postgres arrays')) DESC;
ts_headline — highlight matches
PostgreSQL
SELECT
  title,
  ts_headline(
    'english',
    body,
    websearch_to_tsquery('postgres database'),
    'MaxWords=25, MinWords=10, StartSel=<b>, StopSel=</b>'
  ) AS excerpt
FROM posts
WHERE search_vec @@ websearch_to_tsquery('postgres database');

Stored procedure (with OUT params)
PostgreSQL
CREATE OR REPLACE PROCEDURE
  transfer_funds(
    from_id INT,
    to_id   INT,
    amount  NUMERIC
  )
LANGUAGE plpgsql AS $$
BEGIN
  UPDATE accounts SET balance = balance - amount WHERE id = from_id;
  UPDATE accounts SET balance = balance + amount WHERE id = to_id;

  IF (SELECT balance FROM accounts WHERE id = from_id) < 0 THEN
    RAISE EXCEPTION 'Insufficient funds';
  END IF;
END;
$$;

CALL transfer_funds(1, 2, 100.00);

Views + Materialized Views

Regular views, updatable views, materialized views with refresh.

Regular view
Both
CREATE OR REPLACE VIEW customer_summary AS
SELECT
  u.id,
  u.name,
  u.email,
  COUNT(o.id)    AS order_count,
  SUM(o.total)   AS lifetime_value,
  MAX(o.created_at) AS last_order
FROM users u
LEFT JOIN orders o ON o.user_id = u.id
GROUP BY u.id, u.name, u.email;

SELECT * FROM customer_summary
WHERE lifetime_value > 500;
Materialized view (PostgreSQL only)
PostgreSQL
-- Creates and caches the result on disk
CREATE MATERIALIZED VIEW daily_revenue AS
SELECT
  DATE(created_at) AS day,
  COUNT(*)         AS orders,
  SUM(total)       AS revenue
FROM orders
GROUP BY DATE(created_at);

-- Add indexes on the materialized view
CREATE INDEX ON daily_revenue (day);

-- Refresh (blocks reads)
REFRESH MATERIALIZED VIEW daily_revenue;

-- Concurrent refresh (no read blocking, needs UNIQUE index)
CREATE UNIQUE INDEX ON daily_revenue (day);
REFRESH MATERIALIZED VIEW CONCURRENTLY daily_revenue;
Materialized views store results physically — great for expensive aggregations. MySQL has no equivalent; you'd need an application-level cache table.

LISTEN / NOTIFY

Real-time pub/sub messaging between database connections.

No extra infrastructure needed. LISTEN/NOTIFY lets you build real-time features (live dashboards, cache invalidation, job queues) using only PostgreSQL — no Redis, Kafka, or message broker required for simple cases.
Basic LISTEN / NOTIFY
PostgreSQL
-- Connection A: subscribe to channel
LISTEN new_order;

-- Connection B: publish event with payload
NOTIFY new_order, '{"order_id":42,"total":199.99}';

-- Connection A receives async notification immediately
-- Payload is a text string (use JSON for structured data)
Trigger-based notification
PostgreSQL
CREATE OR REPLACE FUNCTION notify_new_order()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  PERFORM pg_notify(
    'new_order',
    row_to_json(NEW)::text
  );
  RETURN NEW;
END; $$;

CREATE TRIGGER trg_notify_order
AFTER INSERT ON orders
FOR EACH ROW
EXECUTE FUNCTION notify_new_order();

-- Any app listening on 'new_order' channel
-- gets notified instantly on each INSERT
Node.js example
Node.js + pg
import { Client } from 'pg';

const client = new Client();
await client.connect();
await client.query('LISTEN new_order');

client.on('notification', (msg) => {
  const order = JSON.parse(msg.payload);
  console.log('New order:', order.order_id);
  // update dashboard, send email, etc.
});

LATERAL Joins

Reference outer query rows from a subquery in the FROM clause.

Top N per group — most common use case
PostgreSQL
-- Get the 3 most recent orders per user
SELECT u.name, recent.id, recent.total, recent.created_at
FROM users u
JOIN LATERAL (
  SELECT id, total, created_at
  FROM orders o
  WHERE o.user_id = u.id     -- ← references outer row
  ORDER BY created_at DESC
  LIMIT 3
) recent ON TRUE
ORDER BY u.name, recent.created_at DESC;
Without LATERAL you'd need a window function + subquery. LATERAL makes this natural and often faster.
LATERAL with LEFT JOIN (include users with no orders)
PostgreSQL
SELECT u.name, recent.total
FROM users u
LEFT JOIN LATERAL (
  SELECT total
  FROM orders o
  WHERE o.user_id = u.id
  ORDER BY created_at DESC
  LIMIT 1
) recent ON TRUE;
-- Users with no orders: recent.total = NULL
LATERAL with UNNEST / set-returning functions
PostgreSQL
-- Expand each user's tags array into rows
SELECT u.name, tag
FROM users u,
LATERAL unnest(u.tags) AS tag;

-- Same as above but explicit CROSS JOIN LATERAL:
SELECT u.name, tag
FROM users u
CROSS JOIN LATERAL unnest(u.tags) AS tag;

GENERATE_SERIES — date/number sequences
PostgreSQL
-- Generate a series of dates (fill gaps)
SELECT
  day::date,
  COALESCE(daily.revenue, 0) AS revenue
FROM
  generate_series(
    '2024-01-01'::date,
    '2024-12-31'::date,
    '1 day'::interval
  ) AS day
LEFT JOIN (
  SELECT DATE(created_at) d, SUM(total) revenue
  FROM orders GROUP BY d
) daily ON daily.d = day::date;
Great for building time-series charts with no gaps — the LEFT JOIN fills missing days with 0.

Dump + Restore

Backup, restore, partial dump, copy between databases.

MySQL — mysqldump 
# Dump entire database
mysqldump -u root -p mydb > backup.sql

# Dump with create-database statement
mysqldump -u root -p --databases mydb > backup_with_create.sql

# Dump all databases
mysqldump -u root -p --all-databases > all_dbs.sql

# Dump only specific tables
mysqldump -u root -p mydb users orders > tables.sql

# Dump schema only (no data)
mysqldump -u root -p --no-data mydb > schema_only.sql

# Dump data only (no schema)
mysqldump -u root -p --no-create-info mydb > data_only.sql

# Compressed dump
mysqldump -u root -p mydb | gzip > backup.sql.gz

# Restore
mysql -u root -p mydb < backup.sql
mysql -u root -p mydb < backup.sql.gz   # won't work, decompress first
gunzip < backup.sql.gz | mysql -u root -p mydb

# Dump with WHERE filter
mysqldump -u root -p mydb orders \
  --where="created_at > '2024-01-01'" > recent_orders.sql
PostgreSQL — pg_dump / pg_restore 
# Plain SQL dump
pg_dump -U postgres mydb > backup.sql

# Custom format (compressed, supports parallel restore)
pg_dump -U postgres -Fc mydb > backup.dump

# Directory format (parallel dump)
pg_dump -U postgres -Fd -j 4 mydb -f backup_dir/

# Schema only
pg_dump -U postgres --schema-only mydb > schema.sql

# Data only
pg_dump -U postgres --data-only mydb > data.sql

# Specific tables
pg_dump -U postgres -t users -t orders mydb > tables.sql

# Exclude a table
pg_dump -U postgres --exclude-table=logs mydb > no_logs.sql

# All databases (roles + tablespaces too)
pg_dumpall -U postgres > all_dbs.sql

# Restore SQL format
psql -U postgres mydb < backup.sql

# Restore custom format
pg_restore -U postgres -d mydb backup.dump

# Parallel restore (4 workers)
pg_restore -U postgres -d mydb -j 4 backup.dump

# Restore only specific table
pg_restore -U postgres -d mydb -t users backup.dump
COPY — fast bulk load/export
PostgreSQL (server-side)
-- Export table to CSV (server writes file)
COPY users TO '/tmp/users.csv'
  WITH (FORMAT CSV, HEADER, DELIMITER ',');

-- Import from CSV
COPY users (name, email) FROM '/tmp/users.csv'
  WITH (FORMAT CSV, HEADER);

-- Client-side (\copy in psql — no server file access needed)
\copy users TO 'users.csv' CSV HEADER
\copy users FROM 'users.csv' CSV HEADER

-- COPY from SELECT (export filtered data)
COPY (
  SELECT * FROM users WHERE created_at > '2024-01-01'
) TO '/tmp/new_users.csv' CSV HEADER;
COPY is orders of magnitude faster than row-by-row INSERTs for bulk loading. MySQL equivalent is LOAD DATA INFILE.
MySQL: LOAD DATA INFILE
MySQL
-- Export to CSV
SELECT * FROM users
  INTO OUTFILE '/tmp/users.csv'
  FIELDS TERMINATED BY ','
  ENCLOSED BY '"'
  LINES TERMINATED BY '\n';

-- Import from CSV
LOAD DATA INFILE '/tmp/users.csv'
  INTO TABLE users
  FIELDS TERMINATED BY ','
  ENCLOSED BY '"'
  LINES TERMINATED BY '\n'
  IGNORE 1 ROWS;   -- skip header

Streaming replication, logical replication, configuration snippets.

PostgreSQL streaming replication setup
postgresql.conf (primary)
# Enable WAL streaming
wal_level = replica
max_wal_senders = 5
wal_keep_size = 256MB
synchronous_commit = on          # or 'remote_apply' for stronger
synchronous_standby_names = '*' # for sync replication
pg_hba.conf (primary)
host  replication  replica_user  10.0.0.2/32  md5
Shell (create replica)
# Base backup from primary
pg_basebackup -h primary_host -U replica_user \
  -D /var/lib/postgresql/data -P -Xs -R
# -R creates standby.signal + primary_conninfo automatically
PostgreSQL logical replication
PostgreSQL
-- On publisher (source)
-- postgresql.conf: wal_level = logical

CREATE PUBLICATION my_pub
  FOR TABLE users, orders;

-- On subscriber (destination)
CREATE SUBSCRIPTION my_sub
  CONNECTION 'host=primary_host dbname=mydb user=replicator'
  PUBLICATION my_pub;

-- Check subscription status
SELECT * FROM pg_stat_subscription;
Logical replication lets you replicate specific tables, replicate to different PostgreSQL versions, and replicate to other databases (Debezium, etc.).
MySQL replication setup
my.cnf (primary)
[mysqld]
server-id   = 1
log-bin     = mysql-bin
binlog-format = ROW
gtid-mode   = ON
enforce-gtid-consistency = ON

-- Create replication user on primary
CREATE USER 'replica'@'%'
  IDENTIFIED WITH mysql_native_password
  BY 'secret';
GRANT REPLICATION SLAVE ON *.* TO 'replica'@'%';

  SOURCE_HOST = 'primary_host',
  SOURCE_USER = 'replica',
  SOURCE_PASSWORD = 'secret',
  SOURCE_AUTO_POSITION = 1;

Performance Tuning

Config, vacuum, autovacuum, query analysis.

Key PostgreSQL config params
postgresql.conf
# Memory
shared_buffers         = 4GB      # ~25% of total RAM
effective_cache_size   = 12GB     # ~75% of RAM (planner hint)
work_mem               = 64MB     # per sort/hash op — be careful
maintenance_work_mem   = 1GB      # for VACUUM, CREATE INDEX

# Parallelism
max_parallel_workers_per_gather = 4

# WAL / checkpoints

# Logging slow queries

VACUUM + ANALYZE
PostgreSQL
-- Reclaim dead tuple space
VACUUM orders;

-- Full vacuum (locks table, rewrites heap)
VACUUM FULL orders;

-- Update planner statistics
ANALYZE orders;

-- Both at once
VACUUM ANALYZE orders;

-- Check table bloat
SELECT
  relname,
  n_dead_tup,
  n_live_tup,
  last_vacuum,
  last_autovacuum
FROM pg_stat_user_tables
ORDER BY n_dead_tup DESC;
Find missing indexes — table stats
PostgreSQL
-- Tables with lots of sequential scans (may need indexes)
SELECT
  relname,
  seq_scan,
  seq_tup_read,
  idx_scan,
  idx_tup_fetch
FROM pg_stat_user_tables
WHERE seq_scan > 100
ORDER BY seq_tup_read DESC;

-- Unused indexes (wasting write performance)
SELECT
  indexrelname,
  idx_scan,
  idx_tup_read
FROM pg_stat_user_indexes
WHERE idx_scan = 0
  AND indexrelname NOT LIKE '%pkey%';
